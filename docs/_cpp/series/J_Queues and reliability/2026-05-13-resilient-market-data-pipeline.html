What this builds

An end-to-end market data pipeline applying every reliability pattern from this series: circuit breaker on the feed connection, bounded queue with backpressure, watchdog on the pricer thread, graceful degradation to stale prices when the feed is unavailable, and health checks for the whole system.

Architecture

Feed (TCP) -> FeedHandler -> BoundedQueue -> Pricer -> RiskEngine
                 |                |              |
           CircuitBreaker    Backpressure    Watchdog
           Reconnect loop    Drop counter    Heartbeat

Feed handler with circuit breaker and reconnect

class FeedHandler {
    std::atomic<bool>    running_{true};
    CircuitBreaker        breaker_{5, 30.0};  // 5 failures, 30s timeout
    BoundedQueue<Tick>    queue_{65536};
    std::atomic<uint64_t> dropped_{0};
    std::atomic<int64_t>  last_tick_ns_{0};

public:
    void run() {
        double delay_s = 1.0;
        while (running_.load()) {
            try {
                breaker_.call([&] { connect_and_stream(); });
                delay_s = 1.0;
            } catch (const std::exception& e) {
                delay_s = std::min(delay_s * 2.0, 60.0);
                double jitter = delay_s * (0.5 + 0.5 * rand() / RAND_MAX);
                std::this_thread::sleep_for(std::chrono::duration<double>(jitter));
            }
        }
    }

    void connect_and_stream() {
        auto conn = tcp_connect("feed:5555");
        while (running_.load()) {
            auto tick = conn.recv_tick();
            last_tick_ns_.store(
                std::chrono::steady_clock::now().time_since_epoch().count(),
                std::memory_order_relaxed);
            if (!queue_.push(tick, std::chrono::milliseconds(0)))
                dropped_.fetch_add(1, std::memory_order_relaxed);
        }
    }

    bool is_fresh(std::chrono::seconds max_age = std::chrono::seconds(60)) const {
        auto now  = std::chrono::steady_clock::now().time_since_epoch().count();
        auto last = last_tick_ns_.load(std::memory_order_relaxed);
        return (now - last) < std::chrono::nanoseconds(max_age).count();
    }
    uint64_t dropped() const { return dropped_.load(); }
};

Pricer thread with watchdog and price fallback

class PricerThread {
    BoundedQueue<Tick>&       feed_q_;
    BoundedQueue<PriceResult> risk_q_{1024};
    Watchdog                  watchdog_{std::chrono::seconds(10)};
    PriceCache                cache_;

public:
    void run() {
        while (running_.load()) {
            auto tick = feed_q_.pop(std::chrono::milliseconds(100));
            if (!tick) { watchdog_.heartbeat(); continue; }
            try {
                auto result = price_tick(*tick);
                cache_.update(tick->symbol, result.price);
                risk_q_.push(result, std::chrono::milliseconds(10));
            } catch (const std::exception& e) {
                auto stale = cache_.get(tick->symbol);
                if (stale) risk_q_.push({tick->symbol, *stale, true, "stale"},
                                        std::chrono::milliseconds(10));
            }
            watchdog_.heartbeat();
        }
    }
};

Rust resilient market data pipeline

use std::sync::atomic::{AtomicBool, AtomicU64, Ordering};
use std::sync::Arc;
use std::time::Duration;
use tokio::sync::mpsc;
use tokio::time::{sleep, Instant};

struct FeedHandler {
    running:    Arc<AtomicBool>,
    dropped:    Arc<AtomicU64>,
    last_tick:  Arc<AtomicU64>, // epoch nanos
    tx:         mpsc::Sender<Tick>,
}

impl FeedHandler {
    fn new(tx: mpsc::Sender<Tick>) -> Self {
        Self {
            running: Arc::new(AtomicBool::new(true)),
            dropped: Arc::new(AtomicU64::new(0)),
            last_tick: Arc::new(AtomicU64::new(0)),
            tx,
        }
    }

    async fn run(&self) {
        let mut delay = Duration::from_secs(1);
        let breaker = CircuitBreaker::new(5, 30.0);

        while self.running.load(Ordering::Relaxed) {
            match breaker.call(|| self.connect_and_stream()).await {
                Ok(_) => delay = Duration::from_secs(1),
                Err(e) => {
                    let jitter = delay.mul_f64(0.5 + rand::random::<f64>());
                    eprintln!("Feed error: {}. Reconnecting in {:?}", e, jitter);
                    sleep(jitter).await;
                    delay = delay.mul_f64(2.0).min(Duration::from_secs(60));
                }
            }
        }
    }

    async fn connect_and_stream(&self) -> Result<(), Box<dyn std::error::Error>> {
        let conn = tcp_connect("feed:5555").await?;
        while self.running.load(Ordering::Relaxed) {
            let tick = conn.recv_tick().await?;
            self.last_tick.store(
                Instant::now().elapsed().as_nanos() as u64, Ordering::Relaxed);
            if self.tx.try_send(tick).is_err() {
                self.dropped.fetch_add(1, Ordering::Relaxed);
            }
        }
        Ok(())
    }
}

// Pricer task with watchdog and fallback
struct PricerTask {
    rx:        mpsc::Receiver<Tick>,
    risk_tx:   mpsc::Sender<PriceResult>,
    cache:     HashMap<String, f64>,
    heartbeat: Arc<AtomicU64>,
}

impl PricerTask {
    async fn run(&mut self) {
        loop {
            match tokio::time::timeout(Duration::from_millis(100), self.rx.recv()).await {
                Ok(Some(tick)) => {
                    let result = match price_tick(&tick) {
                        Ok(r) => {
                            self.cache.insert(tick.symbol.clone(), r.price);
                            r
                        }
                        Err(_) => {
                            // Fallback to cached price
                            if let Some(&stale) = self.cache.get(&tick.symbol) {
                                PriceResult {
                                    symbol: tick.symbol, price: stale,
                                    is_stale: true, source: "stale".into(),
                                }
                            } else { continue; }
                        }
                    };
                    let _ = self.risk_tx.try_send(result);
                }
                Ok(None) => break, // channel closed
                Err(_) => {} // timeout — just heartbeat
            }
            self.heartbeat.store(
                Instant::now().elapsed().as_nanos() as u64, Ordering::Relaxed);
        }
    }
}
Rust's mpsc::channel replaces the hand-rolled bounded queue — backpressure via try_send(), capacity set at creation. The tokio::time::timeout wrapper on recv() doubles as a heartbeat mechanism: even if no ticks arrive, the pricer loop continues and updates its heartbeat.

Go resilient market data pipeline

package main

import (
    "context"
    "log"
    "math/rand"
    "sync/atomic"
    "time"
)

type FeedHandler struct {
    feed     chan Tick
    dropped  atomic.Uint64
    lastTick atomic.Int64 // unix nano
    breaker  *CircuitBreaker
}

func NewFeedHandler(bufSize int) *FeedHandler {
    return &FeedHandler{
        feed:    make(chan Tick, bufSize),
        breaker: NewCircuitBreaker(5, 30*time.Second),
    }
}

func (fh *FeedHandler) Run(ctx context.Context) {
    delay := time.Second
    for {
        select {
        case <-ctx.Done():
            close(fh.feed)
            return
        default:
        }

        err := fh.breaker.Call(func() error {
            return fh.connectAndStream(ctx)
        })
        if err != nil {
            jitter := time.Duration(float64(delay) * (0.5 + rand.Float64()))
            log.Printf("Feed error: %v. Reconnecting in %v", err, jitter)
            select {
            case <-ctx.Done():
                return
            case <-time.After(jitter):
            }
            if delay < 60*time.Second {
                delay *= 2
            }
        } else {
            delay = time.Second
        }
    }
}

func (fh *FeedHandler) connectAndStream(ctx context.Context) error {
    conn, err := tcpConnect("feed:5555")
    if err != nil {
        return err
    }
    defer conn.Close()

    for {
        select {
        case <-ctx.Done():
            return nil
        default:
        }
        tick, err := conn.RecvTick()
        if err != nil {
            return err
        }
        fh.lastTick.Store(time.Now().UnixNano())

        select {
        case fh.feed <- tick:
        default:
            fh.dropped.Add(1) // channel full — drop
        }
    }
}

// Pricer goroutine with watchdog and fallback
func pricerLoop(ctx context.Context, feed <-chan Tick, riskCh chan<- PriceResult,
    heartbeat *atomic.Int64) {

    cache := make(map[string]float64)

    for {
        select {
        case <-ctx.Done():
            return
        case tick, ok := <-feed:
            if !ok { return }
            result, err := priceTick(tick)
            if err != nil {
                if stale, ok := cache[tick.Symbol]; ok {
                    result = PriceResult{Symbol: tick.Symbol, Price: stale,
                        IsStale: true, Source: "stale"}
                } else {
                    continue
                }
            } else {
                cache[tick.Symbol] = result.Price
            }
            select {
            case riskCh <- result:
            default: // risk queue full — drop
            }
        case <-time.After(100 * time.Millisecond):
            // no tick — just heartbeat
        }
        heartbeat.Store(time.Now().UnixNano())
    }
}

// Main — wire everything together
func main() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    fh := NewFeedHandler(65536)
    riskCh := make(chan PriceResult, 1024)
    var heartbeat atomic.Int64

    go fh.Run(ctx)
    go pricerLoop(ctx, fh.feed, riskCh, &heartbeat)
    go watchdog(ctx, &heartbeat, 10*time.Second)

    // Risk engine consumes from riskCh
    for result := range riskCh {
        riskEngine.Process(result)
    }
}
The entire pipeline is wired together with channels and goroutines — no thread pool configuration, no executor setup. The context.Context propagates cancellation cleanly from main through the feed handler, pricer, and watchdog. The select with time.After in the pricer doubles as a heartbeat timer. This is where Go's concurrency model truly shines.

TypeScript resilient market data pipeline

// Feed handler with circuit breaker and reconnect
class FeedHandler {
  private dropped = 0;
  private lastTickMs = 0;
  private running = true;
  private breaker = new CircuitBreaker(5, 30_000);

  constructor(private queue: BoundedQueue<Tick>) {}

  async run(): Promise<void> {
    let delayMs = 1000;
    while (this.running) {
      try {
        await this.breaker.call(() => this.connectAndStream());
        delayMs = 1000;
      } catch (err) {
        const jitter = delayMs * (0.5 + Math.random());
        console.error(`Feed error: ${err}. Reconnecting in ${(jitter/1000).toFixed(1)}s`);
        await new Promise(r => setTimeout(r, jitter));
        delayMs = Math.min(delayMs * 2, 60_000);
      }
    }
  }

  private async connectAndStream(): Promise<void> {
    const conn = await tcpConnect("feed:5555");
    while (this.running) {
      const tick = await conn.recvTick();
      this.lastTickMs = Date.now();
      if (!this.queue.tryPush(tick)) {
        this.dropped++;
      }
    }
  }

  isFresh(maxAgeMs = 60_000): boolean {
    return Date.now() - this.lastTickMs < maxAgeMs;
  }

  stop(): void { this.running = false; }
}

// Pricer with fallback
class PricerPipeline {
  private cache = new Map<string, number>();
  private lastHeartbeat = Date.now();

  constructor(
    private feedQueue: BoundedQueue<Tick>,
    private riskQueue: BoundedQueue<PriceResult>,
  ) {}

  async run(): Promise<void> {
    while (true) {
      const tick = this.feedQueue.tryPop();
      if (tick) {
        let result: PriceResult;
        try {
          result = await priceTick(tick);
          this.cache.set(tick.symbol, result.price);
        } catch {
          const stale = this.cache.get(tick.symbol);
          if (stale !== undefined) {
            result = { symbol: tick.symbol, price: stale, isStale: true, source: "stale" };
          } else {
            continue;
          }
        }
        this.riskQueue.tryPush(result);
      }
      this.lastHeartbeat = Date.now();
      await new Promise(r => setTimeout(r, 1)); // yield to event loop
    }
  }
}

// Alerting thresholds
interface AlertConfig {
  feedAgeWarnMs:    number; // 30_000
  feedAgeCritMs:    number; // 60_000
  queueWarnPct:    number; // 0.70
  queueCritPct:    number; // 0.90
}

function startMonitoring(
  feedHandler: FeedHandler,
  feedQueue: BoundedQueue<Tick>,
  config: AlertConfig,
): NodeJS.Timeout {
  return setInterval(() => {
    const queuePct = feedQueue.size / feedQueue.capacity;
    if (queuePct > config.queueCritPct)
      console.error(`CRITICAL: queue at ${(queuePct*100).toFixed(0)}%`);
    else if (queuePct > config.queueWarnPct)
      console.warn(`WARN: queue at ${(queuePct*100).toFixed(0)}%`);

    if (!feedHandler.isFresh(config.feedAgeCritMs))
      console.error("CRITICAL: feed stale");
    else if (!feedHandler.isFresh(config.feedAgeWarnMs))
      console.warn("WARN: feed aging");
  }, 1000);
}
Node.js requires explicit await new Promise(r => setTimeout(r, 1)) to yield to the event loop between ticks. Without it, the pricer loop would starve the health check endpoints and monitoring timers. This is the key architectural difference from Go/Rust where concurrent tasks run on separate OS threads.

Key observations

  • Go wires the entire pipeline in the fewest lines. Channels connect stages, context.Context propagates shutdown, goroutines replace thread management. The select statement handles timeout, cancellation, and data arrival in one construct. For reliability-focused architectures, this is Go's strongest use case.
  • Rust matches C++ on latency with the added benefit of compile-time safety. The mpsc::channel replaces the hand-rolled bounded queue, and tokio::time::timeout wrapping recv() is cleaner than manual condition variable timeouts.
  • C++ remains the choice for ultra-low-latency hot paths where lock-free SPSC queues and manual memory layout matter. The tradeoff: more code, more opportunity for bugs, longer development time.
  • TypeScript can implement the pattern but requires explicit event loop yielding and cannot parallelize CPU-bound pricing without worker threads. Best suited for non-latency-sensitive monitoring and alerting layers.

Decision guide

  • Market data pipeline with many feeds: Go — goroutines scale to thousands of concurrent connections with minimal overhead
  • Ultra-low-latency single-feed path: C++ or Rust — lock-free queues, zero-allocation hot path
  • Monitoring and alerting layer: any language — this is not latency-sensitive
  • Prototype or proof of concept: Go — fastest to wire up, easiest to reason about concurrency
