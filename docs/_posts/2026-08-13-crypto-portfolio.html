The approach

A crypto portfolio tracker fetches live prices from CoinGecko's free API, values your holdings, and visualizes the allocation. No API key required.

Fetch prices from CoinGecko

import requests
from dataclasses import dataclass, field


def fetch_crypto_prices(coin_ids: list[str], currency: str = "usd") -> dict:
    """Fetch price, 24h change, and market cap from CoinGecko."""
    url = "https://api.coingecko.com/api/v3/simple/price"
    params = {
        "ids": ",".join(coin_ids),
        "vs_currencies": currency,
        "include_24hr_change": "true",
        "include_market_cap": "true",
    }
    resp = requests.get(url, params=params, timeout=10)
    resp.raise_for_status()
    return resp.json()


# --- demo ---
prices = fetch_crypto_prices(["bitcoin", "ethereum", "solana", "cardano"])
for coin, data in prices.items():
    print(f"{coin:10s} ${data['usd']:>10,.2f}  24h: {data['usd_24h_change']:+.2f}%")

Portfolio valuation

@dataclass
class CryptoPortfolio:
    """Track crypto holdings and their USD value."""
    holdings: dict[str, float] = field(default_factory=dict)  # coin_id -> quantity
    prices: dict[str, dict] | None = None

    def refresh_prices(self) -> None:
        self.prices = fetch_crypto_prices(list(self.holdings.keys()))

    def valuations(self) -> dict[str, float]:
        """USD value of each position."""
        if self.prices is None:
            self.refresh_prices()
        return {
            coin: qty * self.prices[coin]["usd"]
            for coin, qty in self.holdings.items()
        }

    @property
    def total_value(self) -> float:
        return sum(self.valuations().values())

    def allocation(self) -> dict[str, float]:
        """Percentage allocation per coin."""
        vals = self.valuations()
        total = sum(vals.values())
        return {coin: round(v / total * 100, 2) for coin, v in vals.items()}

    def summary(self) -> str:
        vals = self.valuations()
        lines = []
        for coin, val in sorted(vals.items(), key=lambda x: -x[1]):
            qty = self.holdings[coin]
            price = self.prices[coin]["usd"]
            change = self.prices[coin]["usd_24h_change"]
            lines.append(
                f"{coin:12s} {qty:>10.4f} x ${price:>10,.2f} = ${val:>12,.2f}  ({change:+.2f}%)"
            )
        lines.append(f"{'TOTAL':12s} {'':>10s}   {'':>10s}   ${self.total_value:>12,.2f}")
        return "\n".join(lines)


portfolio = CryptoPortfolio(holdings={
    "bitcoin": 0.5,
    "ethereum": 4.0,
    "solana": 25.0,
    "cardano": 5000.0,
})
portfolio.refresh_prices()
print(portfolio.summary())

Allocation pie chart

import plotly.graph_objects as go

vals = portfolio.valuations()
coins = list(vals.keys())
values = list(vals.values())

fig = go.Figure(go.Pie(
    labels=[c.title() for c in coins],
    values=values,
    hole=0.4,
    textinfo="label+percent",
    textposition="outside",
    marker=dict(colors=["#FFA15A", "#636EFA", "#AB63FA", "#00CC96"]),
))
fig.update_layout(
    title="Portfolio Allocation",
    template="plotly_dark",
    margin=dict(t=50, b=20, l=20, r=20),
    showlegend=False,
    annotations=[dict(
        text=f"${portfolio.total_value:,.0f}",
        x=0.5, y=0.5, font_size=18, showarrow=False,
    )],
)
fig.show()

Historical value chart

import yfinance as yf

# yfinance crypto tickers: BTC-USD, ETH-USD, SOL-USD, ADA-USD
yf_map = {
    "bitcoin": ("BTC-USD", 0.5),
    "ethereum": ("ETH-USD", 4.0),
    "solana": ("SOL-USD", 25.0),
    "cardano": ("ADA-USD", 5000.0),
}

fig = go.Figure()
for coin, (yf_ticker, qty) in yf_map.items():
    df = yf.download(yf_ticker, period="6mo", interval="1d", progress=False)
    fig.add_trace(go.Scatter(
        x=df.index,
        y=df["Close"].values.flatten() * qty,
        mode="lines",
        name=coin.title(),
        stackgroup="portfolio",
    ))

fig.update_layout(
    title="Portfolio Value Over Time (stacked)",
    yaxis_title="USD Value",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
    hovermode="x unified",
)
fig.show()

Key takeaways

  • CoinGecko's /simple/price endpoint is free and needs no API key. Rate limit is ~10-30 calls/minute.
  • The stackgroup parameter in Plotly creates a stacked area chart — useful for seeing both total portfolio value and individual contributions over time.
  • Crypto is 24/7, so unlike equities there are no market-hours gaps in the data. Plan your refresh interval accordingly.
