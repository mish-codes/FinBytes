Dates & Datetimes

date - datetime mixing (TypeError in 3.11)

# 3.6: silently works    3.11: TypeError
from datetime import date, datetime

d = date(2024, 1, 1)
dt = datetime(2024, 1, 1, 0, 0)
dt - d  # TypeError in 3.11

# Fix: convert explicitly
dt - datetime.combine(d, datetime.min.time())  # timedelta(0)

datetime.utcnow() deprecated (3.12+)

# Old
from datetime import datetime
now = datetime.utcnow()  # naive UTC — deprecated

# Fix: use timezone-aware
from datetime import datetime, timezone
now = datetime.now(timezone.utc)

Dict Ordering

dict is insertion-ordered (3.7+)

# 3.6: dict order was CPython implementation detail, not guaranteed
# 3.7+: dict preserves insertion order (language spec)

# You can now rely on this:
config = {"host": "localhost", "port": 5432, "db": "quantlab"}
assert list(config.keys()) == ["host", "port", "db"]  # guaranteed

# No longer need OrderedDict for this use case

Type Hints

Built-in generics (3.9+)

# 3.6: need typing imports
from typing import List, Dict, Optional, Tuple

def get_prices(tickers: List[str]) -> Dict[str, float]: ...

# 3.9+: use built-in types directly
def get_prices(tickers: list[str]) -> dict[str, float]: ...

Union with | operator (3.10+)

# 3.6-3.9
from typing import Union, Optional

def get_price(ticker: str) -> Optional[float]: ...
def fetch(source: Union[str, Path]) -> bytes: ...

# 3.10+
def get_price(ticker: str) -> float | None: ...
def fetch(source: str | Path) -> bytes: ...

TypeAlias and type statement (3.10/3.12)

# 3.10
from typing import TypeAlias
PriceMap: TypeAlias = dict[str, float]

# 3.12
type PriceMap = dict[str, float]

String Formatting

f-string improvements (3.8, 3.12)

# 3.8: = for debugging
price = 189.50
print(f"{price=}")           # price=189.5

# 3.12: any expression in f-strings (nested quotes, lambdas)
data = {"AAPL": 189.5}
print(f"{data["AAPL"]:.2f}")  # 189.50 — quotes inside f-string OK in 3.12

Pattern Matching (3.10+)

# Replace long if/elif chains
def classify_spread(bps: float) -> str:
    match bps:
        case x if x < 50:
            return "tight"
        case x if x < 150:
            return "normal"
        case x if x < 300:
            return "wide"
        case _:
            return "distressed"

# Works on types too
def handle_event(event: dict):
    match event:
        case {"type": "trade", "ticker": str(t), "qty": int(q)}:
            process_trade(t, q)
        case {"type": "cancel", "order_id": str(oid)}:
            cancel_order(oid)
        case _:
            log_unknown(event)

Exception Handling

Exception groups (3.11)

# Raise multiple errors at once (useful for validation)
errors = []
if qty <= 0:
    errors.append(ValueError("quantity must be positive"))
if ticker == "":
    errors.append(ValueError("ticker required"))
if errors:
    raise ExceptionGroup("validation failed", errors)

# Catch selectively
try:
    validate(request)
except* ValueError as eg:
    for e in eg.exceptions:
        print(f"Validation error: {e}")

Exception notes (3.11)

try:
    price = get_price("INVALID")
except Exception as e:
    e.add_note(f"Occurred while pricing portfolio P-123")
    e.add_note(f"Timestamp: {datetime.now(timezone.utc)}")
    raise  # notes appear in traceback

Performance (3.11)

CPython 10-60% faster

# No code changes needed — same code runs faster
# Adaptive specialising interpreter (PEP 659)
# Biggest gains: pure Python loops, attribute access, function calls

# Benchmark tip:
import timeit
timeit.timeit("sum(range(1_000_000))", number=100)
# 3.6:  ~4.2s
# 3.11: ~2.8s  (roughly 30% faster on this micro-benchmark)

Async Improvements

asyncio.TaskGroup (3.11)

# 3.6: asyncio.gather — if one task fails, others keep running
results = await asyncio.gather(fetch("AAPL"), fetch("MSFT"))

# 3.11: TaskGroup — structured concurrency, cancels siblings on failure
async with asyncio.TaskGroup() as tg:
    t1 = tg.create_task(fetch("AAPL"))
    t2 = tg.create_task(fetch("MSFT"))
# both done here, or ExceptionGroup raised if any failed

asyncio.Runner (3.11)

# 3.6
asyncio.get_event_loop().run_until_complete(main())

# 3.10+
asyncio.run(main())

# 3.11: Runner for multiple calls in same loop
with asyncio.Runner() as runner:
    runner.run(setup())
    runner.run(main())
    runner.run(cleanup())

Other Breaking Changes

collections.abc moved

# 3.6: works
from collections import Mapping, Sequence

# 3.10+: removed — use collections.abc
from collections.abc import Mapping, Sequence

importlib.resources

# 3.6: pkg_resources (slow, deprecated)
import pkg_resources
data = pkg_resources.resource_string("mypackage", "data.json")

# 3.9+: importlib.resources (fast, stdlib)
from importlib.resources import files
data = files("mypackage").joinpath("data.json").read_text()

ssl defaults tightened

# 3.10+: ssl module rejects older TLS versions by default
# If connecting to legacy systems:
import ssl
ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
ctx.minimum_version = ssl.TLSVersion.TLSv1_2  # explicit minimum

Migration Checklist

  1. Run python -W all to surface DeprecationWarnings
  2. Search for from typing import List, Dict, Tuple, Optional — replace with built-ins
  3. Search for Union[ — replace with |
  4. Search for datetime.utcnow() — replace with datetime.now(timezone.utc)
  5. Search for from collections import (non-abc) — add .abc
  6. Run pyupgrade --py311-plus **/*.py for automated fixes
  7. Run full test suite under 3.11 — watch for date/datetime TypeError
  8. Check CI uses the same Python version as production
