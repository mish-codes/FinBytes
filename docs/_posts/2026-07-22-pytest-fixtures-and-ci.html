Fixtures: Setup That Scales

A pytest fixture is a function that provides data or state to your tests. Instead of repeating setup code in every test, you define it once and pytest injects it by name.

import pytest

@pytest.fixture
def sample_portfolio():
    """Provide a small portfolio for testing."""
    return {
        "AAPL": {"qty": 100, "price": 189.5},
        "MSFT": {"qty": 50, "price": 420.1},
        "GOOG": {"qty": 30, "price": 175.3},
    }

def test_portfolio_total(sample_portfolio):
    total = sum(v["qty"] * v["price"] for v in sample_portfolio.values())
    assert total == pytest.approx(100 * 189.5 + 50 * 420.1 + 30 * 175.3)

def test_portfolio_has_three_positions(sample_portfolio):
    assert len(sample_portfolio) == 3

Setup and teardown with yield

Code before yield runs as setup. Code after yield runs as teardown, even if the test fails.

from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from models import Base

@pytest.fixture
def db_session():
    """Provide a clean in-memory database, torn down after each test."""
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    Session = sessionmaker(bind=engine)
    session = Session()

    yield session          # <-- test runs here

    session.close()        # <-- teardown: always runs
    engine.dispose()

Scope Levels

Scope controls how often the fixture is created and destroyed. Choosing the right scope is the difference between a 5-second test suite and a 5-minute one.

Scope Created Destroyed Use When
function (default) Before each test After each test State must be fresh (DB sessions, mutable data)
class Once per test class After last test in class Expensive setup, tests only read (no mutation)
module Once per .py file After last test in file Docker containers, large data loads
session Once per test run After all tests finish Connection pools, compiled models

Seeing scope in action

import pytest

@pytest.fixture(scope="session")
def db_engine():
    print("\n  [SESSION] Creating engine")
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    yield engine
    print("\n  [SESSION] Disposing engine")
    engine.dispose()

@pytest.fixture(scope="function")
def db_session(db_engine):
    print("\n  [FUNCTION] Opening session")
    Session = sessionmaker(bind=db_engine)
    session = Session()
    yield session
    print("\n  [FUNCTION] Rolling back + closing session")
    session.rollback()
    session.close()

The engine is created once. Each test gets a fresh session that rolls back changes, keeping tests isolated without the overhead of recreating the schema every time.

conftest.py: Shared Fixtures Without Imports

Any fixture defined in conftest.py is automatically available to tests in the same directory (and subdirectories). No import needed.

project/
  conftest.py          # fixtures available to ALL tests
  tests/
    conftest.py        # fixtures available to tests/ and below
    test_scanner.py
    integration/
      conftest.py      # fixtures available only in integration/
      test_db.py

Layering: inner conftest overrides outer

# project/conftest.py
@pytest.fixture
def api_url():
    return "https://api.example.com"

# project/tests/integration/conftest.py
@pytest.fixture
def api_url():
    return "http://localhost:8000"   # override for integration tests

Tests in tests/integration/ get the local URL. Tests in tests/ get the production-like URL. No conditional logic needed.

Parametrize: Data-Driven Tests

@pytest.mark.parametrize

Run the same test logic with different inputs. Each parameter set is a separate test case in the output.

import pytest
from bond_math import present_value

@pytest.mark.parametrize("rate, periods, fv, expected_pv", [
    (0.05, 10, 1000, 613.91),     # 5% rate, 10 years
    (0.03, 5,  1000, 862.61),     # 3% rate, 5 years
    (0.00, 10, 1000, 1000.00),    # 0% rate, no discounting
    (0.10, 1,  1000, 909.09),     # 10% rate, 1 year
])
def test_present_value(rate, periods, fv, expected_pv):
    result = present_value(rate, periods, fv)
    assert result == pytest.approx(expected_pv, rel=1e-2)

@pytest.fixture(params=[...])

Fixture-level parametrization runs all tests that use the fixture once per parameter.

@pytest.fixture(params=["sqlite:///:memory:", "postgresql://test:test@localhost/testdb"])
def db_engine(request):
    """Run tests against both SQLite and PostgreSQL."""
    engine = create_engine(request.param)
    Base.metadata.create_all(engine)
    yield engine
    engine.dispose()

def test_insert_position(db_engine):
    """This test runs twice: once with SQLite, once with PostgreSQL."""
    session = sessionmaker(bind=db_engine)()
    session.add(Position(ticker="AAPL", quantity=100, price=189.5))
    session.commit()
    assert session.query(Position).count() == 1
    session.close()

When to use which: use @pytest.mark.parametrize when varying test inputs. Use @pytest.fixture(params=...) when varying infrastructure (different databases, different API versions, different configs).

CI Gotchas: Tests Pass Locally, Fail in CI

1. Path separators: backslash vs forward slash

# WRONG: hardcoded Windows path — fails on Linux CI
def test_report_path():
    assert report.output_path == "reports\\daily\\var.csv"

# RIGHT: use pathlib (platform-agnostic)
from pathlib import Path

def test_report_path():
    expected = Path("reports") / "daily" / "var.csv"
    assert Path(report.output_path) == expected

# Or use os.sep / forward slashes in assertions
def test_report_path_alt():
    assert report.output_path.replace("\\", "/") == "reports/daily/var.csv"

2. Timezone dependence

# WRONG: depends on local timezone (your machine is EST, CI is UTC)
from datetime import datetime

def test_report_date():
    report = generate_report()
    assert report["date"] == datetime.now().strftime("%Y-%m-%d")  # off by 1 day at 11pm EST

# RIGHT: use UTC explicitly or freeze time
from datetime import datetime, timezone

def test_report_date():
    report = generate_report()
    expected = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    assert report["date"] == expected

# BEST: freeze time so the test is fully deterministic
from freezegun import freeze_time

@freeze_time("2026-07-22 14:00:00", tz_offset=0)
def test_report_date():
    report = generate_report()
    assert report["date"] == "2026-07-22"

3. Import order: editable install vs PYTHONPATH

Locally you run pip install -e . which puts your package on sys.path. In CI, the runner might not do an editable install. Tests fail with ModuleNotFoundError.

# pyproject.toml — the fix used in QuantLab exercises
[tool.pytest.ini_options]
pythonpath = [".", "src"]
testpaths = ["tests"]

This tells pytest to add the project root and src/ to sys.path before running tests. Works identically in CI and locally.

4. Fixture scope leaks

# DANGEROUS: session-scoped mutable fixture
@pytest.fixture(scope="session")
def shared_config():
    return {"max_positions": 100, "risk_limit": 1_000_000}

def test_update_limit(shared_config):
    shared_config["risk_limit"] = 500_000   # mutates the shared object!
    assert shared_config["risk_limit"] == 500_000

def test_default_limit(shared_config):
    # FAILS: risk_limit is still 500_000 from the previous test
    assert shared_config["risk_limit"] == 1_000_000

# FIX: return a fresh copy, or use function scope for mutable state
@pytest.fixture(scope="session")
def _base_config():
    return {"max_positions": 100, "risk_limit": 1_000_000}

@pytest.fixture
def shared_config(_base_config):
    return _base_config.copy()   # fresh copy per test

5. Parallel execution with pytest-xdist

# Run tests across 4 workers
pytest -n 4

Tests run in random order across workers. Problems that surface:

  • Two tests write to the same file — use tmp_path instead of hardcoded paths
  • Two tests share a database table — use unique table names or separate schemas per worker
  • Session-scoped fixtures run once per worker, not once total
# Worker-safe temp directory pattern
@pytest.fixture
def output_dir(tmp_path, worker_id):
    """Each xdist worker gets its own output directory."""
    if worker_id == "master":
        return tmp_path / "output"
    return tmp_path / f"output-{worker_id}"

6. Missing environment variables

# Locally you have .env loaded by direnv or IDE.
# CI has no .env file. Tests crash with KeyError.

# FIX: set defaults in conftest.py
@pytest.fixture(autouse=True)
def set_test_env(monkeypatch):
    """Ensure critical env vars exist in every test, including CI."""
    defaults = {
        "FRED_API_KEY": "test-key",
        "DB_URL": "sqlite:///:memory:",
        "ENVIRONMENT": "test",
        "AWS_DEFAULT_REGION": "us-east-1",
    }
    for key, val in defaults.items():
        monkeypatch.setenv(key, val)

autouse=True means this fixture runs for every test without being explicitly requested. Every test gets a consistent environment.

pytest Configuration: pyproject.toml

The QuantLab project uses pyproject.toml for all pytest configuration. Here is a production-ready setup:

[tool.pytest.ini_options]
# Where to find tests
testpaths = ["tests"]

# Add project root to sys.path (the QuantLab fix)
pythonpath = [".", "src"]

# Default command-line options
addopts = "-v --tb=short -q --strict-markers"

# Register custom markers so typos are caught
markers = [
    "slow: marks tests that take > 5 seconds (deselect with '-m \"not slow\"')",
    "integration: marks tests using fake external services",
    "e2e: marks end-to-end tests requiring real infrastructure",
]

# Minimum pytest version
minversion = "7.0"

# Pattern for test discovery
python_files = ["test_*.py"]
python_classes = ["Test*"]
python_functions = ["test_*"]

# Common commands
pytest                              # run all tests with configured defaults
pytest -m "not slow"                # skip slow tests
pytest -m integration               # run only integration tests
pytest -k "test_var"                # run tests matching "test_var" in name
pytest --co                         # collect only (list tests, don't run)
pytest -x                           # stop on first failure
pytest --lf                         # re-run only last-failed tests
pytest -n auto                      # parallel execution (requires pytest-xdist)

Putting It Together: A Real conftest.py

# tests/conftest.py — from a QuantLab exercise project
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from models import Base

# ---------- Database ----------
@pytest.fixture(scope="session")
def engine():
    """One engine for the entire test run."""
    eng = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(eng)
    yield eng
    eng.dispose()

@pytest.fixture
def db_session(engine):
    """Fresh session per test, rolled back after."""
    connection = engine.connect()
    transaction = connection.begin()
    session = sessionmaker(bind=connection)()
    yield session
    session.close()
    transaction.rollback()
    connection.close()

# ---------- Environment ----------
@pytest.fixture(autouse=True)
def test_env(monkeypatch):
    monkeypatch.setenv("ENVIRONMENT", "test")
    monkeypatch.setenv("DB_URL", "sqlite:///:memory:")

# ---------- Sample data ----------
@pytest.fixture
def price_series():
    return [100.0, 102.0, 99.5, 101.3, 103.8, 98.7, 105.2]

@pytest.fixture
def portfolio_positions():
    return [
        {"ticker": "AAPL", "qty": 100, "sector": "tech"},
        {"ticker": "JPM",  "qty": 200, "sector": "financials"},
        {"ticker": "XOM",  "qty": 150, "sector": "energy"},
    ]

This conftest gives every test file access to a database session, environment variables, and sample data, with zero imports and full isolation between tests.
