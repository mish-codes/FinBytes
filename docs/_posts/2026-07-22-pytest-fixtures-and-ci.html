Fixtures

Basic fixture

import pytest

@pytest.fixture
def sample_portfolio():
    return {
        "AAPL": {"qty": 100, "price": 189.5},
        "MSFT": {"qty": 50, "price": 420.1},
        "GOOG": {"qty": 30, "price": 175.3},
    }

def test_portfolio_total(sample_portfolio):
    total = sum(v["qty"] * v["price"] for v in sample_portfolio.values())
    assert total == pytest.approx(100 * 189.5 + 50 * 420.1 + 30 * 175.3)

def test_portfolio_count(sample_portfolio):
    assert len(sample_portfolio) == 3

Fixture with teardown (yield)

import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@pytest.fixture
def db_session():
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    session = sessionmaker(bind=engine)()

    yield session          # test runs here

    session.close()        # teardown: always runs
    engine.dispose()

Scope levels

Scope Created Destroyed Use When
function Each test After each test Mutable state, DB sessions
class Once per class After last test in class Expensive setup, read-only
module Once per .py file After last test in file Docker containers
session Once per run After all tests Connection pools 
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@pytest.fixture(scope="session")
def db_engine():
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    yield engine
    engine.dispose()

@pytest.fixture(scope="function")
def db_session(db_engine):
    session = sessionmaker(bind=db_engine)()
    yield session
    session.rollback()
    session.close()

conftest.py

Shared fixtures

# tests/conftest.py — available to all tests, no import needed
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@pytest.fixture(scope="session")
def engine():
    eng = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(eng)
    yield eng
    eng.dispose()

@pytest.fixture
def db_session(engine):
    conn = engine.connect()
    txn = conn.begin()
    session = sessionmaker(bind=conn)()
    yield session
    session.close()
    txn.rollback()
    conn.close()

@pytest.fixture
def sample_prices():
    return {"AAPL": [189.5, 190.1, 188.3], "MSFT": [420.1, 418.5, 422.0]}

Override per directory

Inner conftest.py fixtures override outer ones by name. Tests in tests/integration/ get the local URL; tests elsewhere get the default.

# tests/conftest.py
@pytest.fixture
def api_url():
    return "https://api.example.com"

# tests/integration/conftest.py
@pytest.fixture
def api_url():
    return "http://localhost:8000"

Parametrize

Data-driven tests

import pytest

def present_value(rate, periods, fv):
    return fv / (1 + rate) ** periods

@pytest.mark.parametrize("rate, periods, fv, expected", [
    (0.05, 10, 1000, 613.91),
    (0.03, 5,  1000, 862.61),
    (0.00, 10, 1000, 1000.00),
    (0.10, 1,  1000, 909.09),
])
def test_present_value(rate, periods, fv, expected):
    assert present_value(rate, periods, fv) == pytest.approx(expected, rel=1e-2)

Fixture params

import pytest
from sqlalchemy import create_engine

@pytest.fixture(params=["sqlite:///:memory:", "postgresql://test:test@localhost/testdb"])
def db_engine(request):
    engine = create_engine(request.param)
    Base.metadata.create_all(engine)
    yield engine
    engine.dispose()

def test_insert_position(db_engine):
    session = sessionmaker(bind=db_engine)()
    session.add(Position(ticker="AAPL", quantity=100, price=189.5))
    session.commit()
    assert session.query(Position).count() == 1
    session.close()

CI Gotchas

  1. Path separators. Use pathlib.Path("reports") / "daily" / "var.csv" instead of hardcoded slashes. Backslash paths break on Linux CI runners.
  2. Timezone. Your machine is EST, CI is UTC. Use datetime.now(timezone.utc) or freezegun to pin time in tests.
  3. Import order (pythonpath fix). Locally pip install -e . works; CI may not. Add pythonpath = [".", "src"] to pyproject.toml.
  4. Missing env vars. Use an autouse=True fixture in conftest that calls monkeypatch.setenv for every required variable.
  5. Fixture scope leaks. Session-scoped mutable fixtures share state across tests. Return .copy() from a function-scoped wrapper instead.
  6. Parallel execution. With pytest-xdist, session fixtures run once per worker. Use tmp_path instead of hardcoded file paths to avoid collisions.

pyproject.toml Config

[tool.pytest.ini_options]
testpaths = ["tests"]
pythonpath = [".", "src"]
addopts = "-v --tb=short -q --strict-markers"
markers = [
    "slow: marks tests that take > 5 seconds",
    "integration: marks tests using fake external services",
    "e2e: marks end-to-end tests requiring real infrastructure",
]
minversion = "7.0"
python_files = ["test_*.py"]
python_classes = ["Test*"]
python_functions = ["test_*"]

pytest                        # run all
pytest -m "not slow"          # skip slow
pytest -k "test_var"          # name match
pytest -x                     # stop on first failure
pytest --lf                   # re-run last failed
pytest -n auto                # parallel (pytest-xdist)
