Why cluster customers

Banks and fintechs serve millions of customers with different financial behaviors. Clustering groups similar customers together without labeled data (unsupervised learning). The output drives targeted products: high-income low-spenders get investment offers, high-spenders with low credit get financial literacy nudges.

Synthetic customer data

import numpy as np
import pandas as pd

np.random.seed(42)
n = 500

# Create 4 natural clusters
clusters_params = [
    {"income": (35000, 8000), "spending": (800, 200), "credit": (580, 40)},   # low income, low credit
    {"income": (70000, 12000), "spending": (2500, 500), "credit": (700, 30)},  # middle class
    {"income": (120000, 20000), "spending": (1500, 400), "credit": (760, 25)}, # high income, savers
    {"income": (110000, 25000), "spending": (5000, 800), "credit": (720, 35)}, # high income, spenders
]

dfs = []
for i, params in enumerate(clusters_params):
    size = n // len(clusters_params)
    cluster_df = pd.DataFrame({
        "income": np.random.normal(*params["income"], size),
        "monthly_spending": np.random.normal(*params["spending"], size),
        "credit_score": np.random.normal(*params["credit"], size).clip(300, 850),
        "true_cluster": i,
    })
    dfs.append(cluster_df)

df = pd.concat(dfs, ignore_index=True)
df = df.sample(frac=1, random_state=42).reset_index(drop=True)

print(f"Dataset: {df.shape}")
print(df[["income", "monthly_spending", "credit_score"]].describe().round(0))

Preprocessing

from sklearn.preprocessing import StandardScaler

features = ["income", "monthly_spending", "credit_score"]
X = df[features].values

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

print(f"Scaled means:  {X_scaled.mean(axis=0).round(4)}")
print(f"Scaled stds:   {X_scaled.std(axis=0).round(4)}")

K-Means uses Euclidean distance. Without scaling, income (tens of thousands) would dominate credit_score (hundreds). StandardScaler puts all features on equal footing.

Elbow method

from sklearn.cluster import KMeans
import plotly.graph_objects as go

inertias = []
K_range = range(2, 10)

for k in K_range:
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    km.fit(X_scaled)
    inertias.append(km.inertia_)

fig = go.Figure(go.Scatter(
    x=list(K_range), y=inertias, mode="lines+markers",
    marker=dict(size=8, color="steelblue"),
    line=dict(width=2),
))
fig.update_layout(
    title="Elbow Method: Optimal Number of Clusters",
    xaxis_title="Number of Clusters (k)",
    yaxis_title="Inertia (within-cluster sum of squares)",
    height=400, width=650, template="plotly_white",
)
fig.show()

Inertia drops sharply as k increases, then flattens. The "elbow" is where adding more clusters gives diminishing returns. Here, k=4 should be the natural choice since we built four groups into the data.

Fit K-Means

k_optimal = 4
kmeans = KMeans(n_clusters=k_optimal, random_state=42, n_init=10)
df["cluster"] = kmeans.fit_predict(X_scaled)

print(f"Cluster sizes:\n{df['cluster'].value_counts().sort_index()}")

Cluster profile table

profile = df.groupby("cluster")[features].agg(["mean", "std"]).round(0)
profile.columns = [f"{feat}_{stat}" for feat, stat in profile.columns]

# Simpler view: just means
means = df.groupby("cluster")[features].mean().round(0)
means["size"] = df.groupby("cluster").size()

print("\nCluster Profiles:")
print(means.to_string())

# Label clusters based on profile
labels = {0: "", 1: "", 2: "", 3: ""}
for idx, row in means.iterrows():
    if row["income"] < 50000:
        labels[idx] = "Budget-conscious"
    elif row["monthly_spending"] > 3500:
        labels[idx] = "High spenders"
    elif row["income"] > 100000 and row["monthly_spending"] < 2000:
        labels[idx] = "Affluent savers"
    else:
        labels[idx] = "Middle market"

df["segment"] = df["cluster"].map(labels)
print("\nSegment names:")
for k, v in labels.items():
    print(f"  Cluster {k}: {v}")

Silhouette score

from sklearn.metrics import silhouette_score

score = silhouette_score(X_scaled, df["cluster"])
print(f"Silhouette score: {score:.3f}")

# Compare across k values
for k in range(2, 8):
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    labels_k = km.fit_predict(X_scaled)
    s = silhouette_score(X_scaled, labels_k)
    print(f"  k={k}: silhouette={s:.3f}")

The silhouette score ranges from -1 to 1. Higher is better. It measures how similar each point is to its own cluster versus the nearest other cluster. Scores above 0.5 indicate well-separated clusters.
