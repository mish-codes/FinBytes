The approach

Technical analysis overlays mathematical indicators on price data to spot trends and reversals. The code below computes four core indicators — SMA/EMA, RSI, MACD, and Bollinger Bands — then stacks them in a multi-panel Plotly chart.

Fetch price data

import yfinance as yf
import pandas as pd
import numpy as np

ticker = "AAPL"
df = yf.download(ticker, period="1y", interval="1d", progress=False)
close = df["Close"].values.flatten()
close_series = pd.Series(close, index=df.index)

Moving averages — SMA and EMA

def sma(series: pd.Series, window: int) -> pd.Series:
    """Simple moving average: equal weight over window."""
    return series.rolling(window).mean()

def ema(series: pd.Series, span: int) -> pd.Series:
    """Exponential moving average: recent prices weighted more."""
    return series.ewm(span=span, adjust=False).mean()

df["SMA_20"] = sma(close_series, 20)
df["SMA_50"] = sma(close_series, 50)
df["EMA_12"] = ema(close_series, 12)
df["EMA_26"] = ema(close_series, 26)

Pattern: when the short-term MA crosses above the long-term MA, it signals bullish momentum (golden cross). The reverse is a death cross.

RSI — Relative Strength Index

def rsi(series: pd.Series, period: int = 14) -> pd.Series:
    """RSI: 0-100 oscillator measuring momentum.
    Above 70 = overbought, below 30 = oversold."""
    delta = series.diff()
    gain = delta.where(delta > 0, 0.0)
    loss = -delta.where(delta < 0, 0.0)
    avg_gain = gain.ewm(alpha=1 / period, min_periods=period).mean()
    avg_loss = loss.ewm(alpha=1 / period, min_periods=period).mean()
    rs = avg_gain / avg_loss
    return 100 - (100 / (1 + rs))

df["RSI"] = rsi(close_series)

MACD — Moving Average Convergence Divergence

def macd(series: pd.Series, fast: int = 12, slow: int = 26, signal: int = 9):
    """Returns MACD line, signal line, and histogram."""
    macd_line = ema(series, fast) - ema(series, slow)
    signal_line = ema(macd_line, signal)
    histogram = macd_line - signal_line
    return macd_line, signal_line, histogram

df["MACD"], df["MACD_Signal"], df["MACD_Hist"] = macd(close_series)

Pattern: when the MACD line crosses above the signal line, it suggests upward momentum. The histogram shows the gap between the two — growing bars mean strengthening trend.

Bollinger Bands

def bollinger_bands(series: pd.Series, window: int = 20, num_std: float = 2.0):
    """Upper band, middle (SMA), lower band."""
    middle = sma(series, window)
    std = series.rolling(window).std()
    upper = middle + num_std * std
    lower = middle - num_std * std
    return upper, middle, lower

df["BB_Upper"], df["BB_Mid"], df["BB_Lower"] = bollinger_bands(close_series)

Stacked dashboard chart

import plotly.graph_objects as go
from plotly.subplots import make_subplots

fig = make_subplots(
    rows=3, cols=1, shared_xaxes=True,
    vertical_spacing=0.04,
    row_heights=[0.5, 0.25, 0.25],
    subplot_titles=["Price + Bollinger Bands", "RSI", "MACD"],
)

# --- Row 1: price + Bollinger ---
fig.add_trace(go.Scatter(
    x=df.index, y=close, mode="lines",
    name="Close", line=dict(color="#636EFA", width=1.5),
), row=1, col=1)
fig.add_trace(go.Scatter(
    x=df.index, y=df["SMA_20"], mode="lines",
    name="SMA 20", line=dict(color="#FFA15A", width=1, dash="dot"),
), row=1, col=1)
fig.add_trace(go.Scatter(
    x=df.index, y=df["BB_Upper"], mode="lines",
    name="BB Upper", line=dict(color="#AB63FA", width=0.8),
), row=1, col=1)
fig.add_trace(go.Scatter(
    x=df.index, y=df["BB_Lower"], mode="lines",
    name="BB Lower", line=dict(color="#AB63FA", width=0.8),
    fill="tonexty", fillcolor="rgba(171,99,250,0.1)",
), row=1, col=1)

# --- Row 2: RSI ---
fig.add_trace(go.Scatter(
    x=df.index, y=df["RSI"], mode="lines",
    name="RSI", line=dict(color="#00CC96", width=1.5),
), row=2, col=1)
fig.add_hline(y=70, line_dash="dash", line_color="#EF553B", row=2, col=1)
fig.add_hline(y=30, line_dash="dash", line_color="#00CC96", row=2, col=1)

# --- Row 3: MACD ---
hist_colors = ["#00CC96" if v >= 0 else "#EF553B" for v in df["MACD_Hist"].fillna(0)]
fig.add_trace(go.Bar(
    x=df.index, y=df["MACD_Hist"], name="Histogram",
    marker_color=hist_colors, showlegend=False,
), row=3, col=1)
fig.add_trace(go.Scatter(
    x=df.index, y=df["MACD"], mode="lines",
    name="MACD", line=dict(color="#636EFA", width=1.2),
), row=3, col=1)
fig.add_trace(go.Scatter(
    x=df.index, y=df["MACD_Signal"], mode="lines",
    name="Signal", line=dict(color="#EF553B", width=1, dash="dash"),
), row=3, col=1)

fig.update_layout(
    title=f"{ticker} — Technical Analysis Dashboard",
    template="plotly_dark",
    height=800,
    margin=dict(t=60, b=40, l=60, r=20),
    hovermode="x unified",
    showlegend=True,
    legend=dict(orientation="h", y=1.02, x=0.5, xanchor="center"),
)
fig.show()

Key takeaways

  • RSI works best in range-bound markets. In strong trends, it can stay overbought or oversold for weeks.
  • MACD is a trend-following indicator — it lags by design. Use it to confirm direction, not to time exact entries.
  • Bollinger Bands measure volatility. A squeeze (bands narrowing) often precedes a big move. Price touching the upper band is not automatically a sell signal.
  • No single indicator is reliable alone. The stacked layout lets you check multiple signals at the same timestamp before making a decision.
