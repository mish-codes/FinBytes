Files

Real files with tmp_path

import csv
import pytest
from pathlib import Path

def save_positions(path: Path, rows: list[dict]) -> int:
    fieldnames = rows[0].keys()
    with open(path, "w", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(rows)
    return len(rows)

def test_save_positions(tmp_path):
    out = tmp_path / "positions.csv"
    rows = [
        {"ticker": "AAPL", "qty": 100, "price": 189.5},
        {"ticker": "MSFT", "qty": 50, "price": 420.1},
    ]
    count = save_positions(out, rows)

    assert count == 2
    assert out.exists()
    with open(out) as f:
        reader = list(csv.DictReader(f))
    assert reader[0]["ticker"] == "AAPL"
    assert reader[1]["price"] == "420.1"

Fake files with mock_open

from unittest.mock import patch, mock_open
import json

def load_config(path):
    with open(path) as f:
        return json.load(f)

def test_load_config():
    fake = '{"db_host": "localhost", "db_port": 5432}'
    with patch("builtins.open", mock_open(read_data=fake)):
        cfg = load_config("config.json")
    assert cfg["db_host"] == "localhost"
    assert cfg["db_port"] == 5432

HTTP APIs

Mock requests.get

import requests
from unittest.mock import patch, MagicMock

def fetch_prices(ticker, days=5):
    resp = requests.get(
        f"https://api.example.com/prices/{ticker}",
        params={"days": days}, timeout=10,
    )
    resp.raise_for_status()
    return resp.json()["prices"]

def _mock_response(status=200, json_data=None, error=None):
    mock = MagicMock()
    mock.status_code = status
    mock.json.return_value = json_data or {}
    if error:
        mock.raise_for_status.side_effect = error
    return mock

@patch("__main__.requests.get")
def test_fetch_success(mock_get):
    mock_get.return_value = _mock_response(json_data={"prices": [189.5, 190.1]})
    assert fetch_prices("AAPL", days=2) == [189.5, 190.1]
    mock_get.assert_called_once()

Mock error responses

import pytest
from unittest.mock import patch

@patch("__main__.requests.get")
def test_404(mock_get):
    mock_get.return_value = _mock_response(
        status=404, error=requests.HTTPError("404")
    )
    with pytest.raises(requests.HTTPError):
        fetch_prices("INVALID")

@patch("__main__.requests.get")
def test_timeout(mock_get):
    mock_get.side_effect = requests.Timeout("timed out")
    with pytest.raises(requests.Timeout):
        fetch_prices("AAPL")

@patch("__main__.requests.get")
def test_connection_error(mock_get):
    mock_get.side_effect = requests.ConnectionError("DNS failure")
    with pytest.raises(requests.ConnectionError):
        fetch_prices("AAPL")

Email / SMTP

Mock SMTP and verify calls

import smtplib
from email.message import EmailMessage
from unittest.mock import patch, MagicMock

def send_alert(recipient, subject, body):
    msg = EmailMessage()
    msg["From"] = "alerts@finbytes.dev"
    msg["To"] = recipient
    msg["Subject"] = subject
    msg.set_content(body)
    with smtplib.SMTP("smtp.example.com", 587) as server:
        server.starttls()
        server.login("alerts@finbytes.dev", "secret")
        server.send_message(msg)

@patch("__main__.smtplib.SMTP")
def test_send_alert(MockSMTP):
    mock_server = MockSMTP.return_value.__enter__.return_value

    send_alert("risk@firm.com", "VaR Breach", "Portfolio p1 exceeded limit")

    MockSMTP.assert_called_once_with("smtp.example.com", 587)
    method_names = [c[0] for c in mock_server.method_calls]
    assert method_names == ["starttls", "login", "send_message"]
    sent_msg = mock_server.send_message.call_args[0][0]
    assert sent_msg["To"] == "risk@firm.com"
    assert sent_msg["Subject"] == "VaR Breach"

Databases

SQLite in-memory

import pytest
from sqlalchemy import create_engine, Column, Integer, String, Float
from sqlalchemy.orm import sessionmaker, declarative_base

Base = declarative_base()

class Position(Base):
    __tablename__ = "positions"
    id = Column(Integer, primary_key=True)
    ticker = Column(String)
    quantity = Column(Integer)
    price = Column(Float)

@pytest.fixture
def db_session():
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    session = sessionmaker(bind=engine)()
    yield session
    session.close()

def test_insert_and_query(db_session):
    db_session.add(Position(ticker="AAPL", quantity=100, price=189.5))
    db_session.commit()
    result = db_session.query(Position).filter_by(ticker="AAPL").first()
    assert result.quantity == 100
    assert result.price == 189.5

def test_empty_table(db_session):
    assert db_session.query(Position).count() == 0

moto for AWS (S3, SQS, DynamoDB)

import boto3
import pytest
from moto import mock_aws

@pytest.fixture
def s3_bucket():
    with mock_aws():
        s3 = boto3.client("s3", region_name="us-east-1")
        s3.create_bucket(Bucket="par-curves")
        yield s3

def test_s3_upload(s3_bucket):
    s3_bucket.put_object(Bucket="par-curves", Key="2026-07-20.json",
                         Body='{"rates": [4.2, 4.35]}')
    obj = s3_bucket.get_object(Bucket="par-curves", Key="2026-07-20.json")
    assert b"4.2" in obj["Body"].read()

@pytest.fixture
def ddb_table():
    with mock_aws():
        ddb = boto3.resource("dynamodb", region_name="us-east-1")
        table = ddb.create_table(
            TableName="credit-spreads",
            KeySchema=[{"AttributeName": "ticker", "KeyType": "HASH"}],
            AttributeDefinitions=[{"AttributeName": "ticker", "AttributeType": "S"}],
            BillingMode="PAY_PER_REQUEST",
        )
        yield table

def test_dynamodb_put_get(ddb_table):
    ddb_table.put_item(Item={"ticker": "AAPL", "spread_bps": 45})
    resp = ddb_table.get_item(Key={"ticker": "AAPL"})
    assert resp["Item"]["spread_bps"] == 45

import fakeredis
import pytest

@pytest.fixture
def redis_client():
    return fakeredis.FakeRedis(decode_responses=True)

def test_cache_hit(redis_client):
    redis_client.set("rate:DGS10", "4.25")
    assert redis_client.get("rate:DGS10") == "4.25"

def test_cache_miss(redis_client):
    assert redis_client.get("rate:NONEXISTENT") is None

def test_ttl(redis_client):
    redis_client.setex("rate:DGS10", 300, "4.25")
    assert 0 < redis_client.ttl("rate:DGS10") <= 300
