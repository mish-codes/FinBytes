The Principle: Mock at the Boundary

Unit tests must be fast, deterministic, and independent of external systems. If your code writes files, sends emails, calls APIs, or queries a database, those are side effects. In a unit test, you replace them with fakes. In an integration test, you use lightweight real substitutes (tmp directories, in-memory databases, local containers).

The boundary is the line between your logic and the outside world. Mock at that line, not deep inside your code.

Testing File I/O

Approach 1: tmp_path (real files, auto-cleaned)

pytest's tmp_path fixture gives you a fresh pathlib.Path directory per test. Files written there are automatically cleaned up.

# report_writer.py
import csv
from pathlib import Path

def save_report(path: Path, rows: list[dict]) -> int:
    """Write portfolio positions to CSV. Returns row count."""
    if not rows:
        raise ValueError("No data to write")
    fieldnames = rows[0].keys()
    with open(path, "w", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(rows)
    return len(rows)

# test_report_writer.py
import csv
import pytest
from report_writer import save_report

def test_save_report_writes_csv(tmp_path):
    out_file = tmp_path / "positions.csv"
    rows = [
        {"ticker": "AAPL", "qty": 100, "price": 189.5},
        {"ticker": "MSFT", "qty": 50, "price": 420.1},
    ]
    count = save_report(out_file, rows)

    assert count == 2
    assert out_file.exists()

    # Read back and verify content
    with open(out_file) as f:
        reader = list(csv.DictReader(f))
    assert len(reader) == 2
    assert reader[0]["ticker"] == "AAPL"
    assert reader[1]["price"] == "420.1"

def test_save_report_empty_raises(tmp_path):
    with pytest.raises(ValueError, match="No data"):
        save_report(tmp_path / "empty.csv", [])

Approach 2: mock_open (no files at all)

When you want to test logic that reads a file without creating a real one:

from unittest.mock import patch, mock_open
import json

def load_config(path: str) -> dict:
    with open(path) as f:
        return json.load(f)

def test_load_config():
    fake_json = '{"db_host": "localhost", "db_port": 5432}'
    with patch("builtins.open", mock_open(read_data=fake_json)):
        cfg = load_config("config.json")
    assert cfg["db_host"] == "localhost"
    assert cfg["db_port"] == 5432

When to use which: use tmp_path when the file format itself is part of the contract (CSV columns, JSON schema). Use mock_open when you only care that your code reads/writes and handles the content correctly, not the filesystem behavior.

Testing Email Sending

Never send real emails from tests. Mock the SMTP client and verify the calls.

# alert_service.py
import smtplib
from email.message import EmailMessage

def send_alert(recipient: str, subject: str, body: str) -> None:
    msg = EmailMessage()
    msg["From"] = "alerts@finbytes.dev"
    msg["To"] = recipient
    msg["Subject"] = subject
    msg.set_content(body)

    with smtplib.SMTP("smtp.example.com", 587) as server:
        server.starttls()
        server.login("alerts@finbytes.dev", "secret")
        server.send_message(msg)

# test_alert_service.py
from unittest.mock import patch, MagicMock, call

@patch("alert_service.smtplib.SMTP")
def test_send_alert_correct_flow(MockSMTP):
    mock_server = MockSMTP.return_value.__enter__.return_value

    send_alert("risk@firm.com", "VaR Breach", "Portfolio p1 exceeded limit")

    # Verify connection target
    MockSMTP.assert_called_once_with("smtp.example.com", 587)

    # Verify call order: starttls -> login -> send_message
    method_names = [c[0] for c in mock_server.method_calls]
    assert method_names == ["starttls", "login", "send_message"]

    # Verify the message content
    sent_msg = mock_server.send_message.call_args[0][0]
    assert sent_msg["To"] == "risk@firm.com"
    assert sent_msg["Subject"] == "VaR Breach"

@patch("alert_service.smtplib.SMTP")
def test_send_alert_to_multiple_recipients(MockSMTP):
    mock_server = MockSMTP.return_value.__enter__.return_value

    recipients = ["risk@firm.com", "manager@firm.com", "cro@firm.com"]
    for r in recipients:
        send_alert(r, "VaR Breach", "Limit exceeded")

    assert mock_server.send_message.call_count == 3

Testing HTTP API Calls

Mocking requests.get

# market_data.py
import requests

def fetch_prices(ticker: str, days: int = 5) -> list[float]:
    """Fetch closing prices from market API."""
    resp = requests.get(
        f"https://api.example.com/prices/{ticker}",
        params={"days": days},
        timeout=10,
    )
    resp.raise_for_status()
    return resp.json()["prices"]

# test_market_data.py
from unittest.mock import patch, MagicMock
import requests
import pytest

def _mock_response(status_code=200, json_data=None, raise_for_status=None):
    """Helper: build a mock Response object."""
    mock_resp = MagicMock()
    mock_resp.status_code = status_code
    mock_resp.json.return_value = json_data or {}
    if raise_for_status:
        mock_resp.raise_for_status.side_effect = raise_for_status
    return mock_resp

@patch("market_data.requests.get")
def test_fetch_prices_success(mock_get):
    mock_get.return_value = _mock_response(
        json_data={"prices": [189.5, 190.1, 188.3]}
    )
    result = fetch_prices("AAPL", days=3)
    assert result == [189.5, 190.1, 188.3]
    mock_get.assert_called_once_with(
        "https://api.example.com/prices/AAPL",
        params={"days": 3},
        timeout=10,
    )

@patch("market_data.requests.get")
def test_fetch_prices_404(mock_get):
    mock_get.return_value = _mock_response(
        status_code=404,
        raise_for_status=requests.HTTPError("404 Not Found"),
    )
    with pytest.raises(requests.HTTPError):
        fetch_prices("INVALID")

@patch("market_data.requests.get")
def test_fetch_prices_timeout(mock_get):
    mock_get.side_effect = requests.Timeout("Connection timed out")
    with pytest.raises(requests.Timeout):
        fetch_prices("AAPL")

@patch("market_data.requests.get")
def test_fetch_prices_connection_error(mock_get):
    mock_get.side_effect = requests.ConnectionError("DNS failure")
    with pytest.raises(requests.ConnectionError):
        fetch_prices("AAPL")

monkeypatch approach (simpler when you don't need call tracking)

def test_fetch_prices_monkeypatch(monkeypatch):
    class FakeResponse:
        status_code = 200
        def json(self):
            return {"prices": [189.5, 190.1]}
        def raise_for_status(self):
            pass

    monkeypatch.setattr("market_data.requests.get", lambda *a, **kw: FakeResponse())
    result = fetch_prices("AAPL")
    assert result == [189.5, 190.1]

Testing Database Calls

SQLite in-memory (fast integration tests)

This is the pattern used throughout QuantLab exercises. Replace PostgreSQL with an in-memory SQLite database for tests:

# conftest.py
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from models import Base  # your SQLAlchemy declarative base

@pytest.fixture
def db_session():
    """Provide a clean in-memory database for each test."""
    engine = create_engine("sqlite:///:memory:")
    Base.metadata.create_all(engine)
    Session = sessionmaker(bind=engine)
    session = Session()
    yield session
    session.close()

# test_positions.py
from models import Position
from services.position_service import get_total_exposure

def test_total_exposure(db_session):
    db_session.add_all([
        Position(ticker="AAPL", quantity=100, price=189.5),
        Position(ticker="MSFT", quantity=50, price=420.1),
    ])
    db_session.commit()

    total = get_total_exposure(db_session)
    assert total == pytest.approx(100 * 189.5 + 50 * 420.1)

def test_empty_portfolio(db_session):
    total = get_total_exposure(db_session)
    assert total == 0.0

Mocking the session (pure unit test)

from unittest.mock import MagicMock

def test_get_positions_unit():

moto for AWS services

The moto library spins up fake AWS services in-process. No network, no credentials, no cost. This pattern appears in QuantLab exercises 10, 15, and 16.

import boto3
import pytest
from moto import mock_aws

@pytest.fixture
def s3_bucket():
    """Create a fake S3 bucket for testing."""
    with mock_aws():
        s3 = boto3.client("s3", region_name="us-east-1")
        s3.create_bucket(Bucket="par-curves")
        yield s3

def test_upload_par_curve(s3_bucket):
    from ingestion import upload_curve
    upload_curve(s3_bucket, "par-curves", "2026-07-20.json", '{"rates": [4.2]}')

    obj = s3_bucket.get_object(Bucket="par-curves", Key="2026-07-20.json")
    body = obj["Body"].read().decode()
    assert '"rates"' in body

@pytest.fixture
def sns_topic():
    """Create a fake SNS topic for alert testing."""
    with mock_aws():
        sns = boto3.client("sns", region_name="us-east-1")
        resp = sns.create_topic(Name="risk-alerts")
        topic_arn = resp["TopicArn"]
        yield sns, topic_arn

def test_publish_alert(sns_topic):
    sns, topic_arn = sns_topic
    from alerts import publish_risk_alert
    publish_risk_alert(sns, topic_arn, "VaR breach: portfolio p1")
    # moto tracks published messages internally

# DynamoDB example (QuantLab Ex 16 pattern)
@pytest.fixture
def ddb_table():
    with mock_aws():
        ddb = boto3.resource("dynamodb", region_name="us-east-1")
        table = ddb.create_table(
            TableName="credit-spreads",
            KeySchema=[{"AttributeName": "ticker", "KeyType": "HASH"}],
            AttributeDefinitions=[{"AttributeName": "ticker", "AttributeType": "S"}],
            BillingMode="PAY_PER_REQUEST",
        )
        yield table

def test_store_spread(ddb_table):
    ddb_table.put_item(Item={"ticker": "AAPL", "spread_bps": 45})
    resp = ddb_table.get_item(Key={"ticker": "AAPL"})
    assert resp["Item"]["spread_bps"] == 45

fakeredis for Redis

From the QuantLab exercise 17 caching pattern:

import fakeredis
import pytest

@pytest.fixture
def redis_client():
    """In-memory Redis replacement. No server needed."""
    return fakeredis.FakeRedis(decode_responses=True)

def test_cache_spot_rate(redis_client):
    from cache_layer import cache_rate, get_cached_rate

    cache_rate(redis_client, "DGS10", 4.25)
    result = get_cached_rate(redis_client, "DGS10")
    assert result == 4.25

def test_cache_miss_returns_none(redis_client):
    from cache_layer import get_cached_rate

    result = get_cached_rate(redis_client, "NONEXISTENT")
    assert result is None

def test_cache_ttl(redis_client):
    redis_client.setex("rate:DGS10", 300, "4.25")
    ttl = redis_client.ttl("rate:DGS10")
    assert 0 < ttl <= 300

The Testing Pyramid

Layer Count Speed What's Real Example
Unit Many (70-80%) < 1ms each Nothing external Mock fetch_prices, test scan logic
Integration Some (15-25%) 10-500ms each DB, files, fake AWS SQLite + moto + fakeredis
E2E Few (5-10%) Seconds Everything Hit real API, real DB, verify output

Tag slow tests so you can skip them during development:

import pytest

@pytest.mark.slow
def test_full_pipeline_real_db():
    """Integration test that spins up a real PostgreSQL container."""
    ...

@pytest.mark.integration
def test_s3_upload_with_moto():
    """Uses moto - faster than real AWS but slower than pure unit tests."""
    ...

# pyproject.toml
[tool.pytest.ini_options]
markers = [
    "slow: marks tests that take > 5 seconds",
    "integration: marks tests using fake external services",
]

# Run only fast unit tests
pytest -m "not slow and not integration"

# Run everything
pytest

The goal is a test suite that runs in under 10 seconds for the common case (unit tests only), with the option to run the full suite in CI where speed matters less than coverage.
