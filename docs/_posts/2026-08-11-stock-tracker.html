The approach

A stock tracker needs a candlestick chart with volume, key metrics at a glance, and a 52-week range indicator. yfinance provides the data, plotly.graph_objects builds the interactive charts.

Fetch stock data

import yfinance as yf
import pandas as pd
from dataclasses import dataclass


@dataclass
class StockSnapshot:
    """Key metrics for a single ticker."""
    ticker: str
    price: float
    change_pct: float
    market_cap: str
    pe_ratio: float | None
    week52_low: float
    week52_high: float
    volume: int

    @classmethod
    def from_yfinance(cls, ticker: str) -> "StockSnapshot":
        t = yf.Ticker(ticker)
        info = t.info
        prev = info.get("previousClose", 1)
        price = info.get("currentPrice", info.get("regularMarketPrice", 0))
        return cls(
            ticker=ticker,
            price=price,
            change_pct=round((price / prev - 1) * 100, 2),
            market_cap=cls._fmt_cap(info.get("marketCap", 0)),
            pe_ratio=info.get("trailingPE"),
            week52_low=info.get("fiftyTwoWeekLow", 0),
            week52_high=info.get("fiftyTwoWeekHigh", 0),
            volume=info.get("volume", 0),
        )

    @staticmethod
    def _fmt_cap(cap: int) -> str:
        if cap >= 1e12:
            return f"${cap / 1e12:.2f}T"
        if cap >= 1e9:
            return f"${cap / 1e9:.2f}B"
        return f"${cap / 1e6:.0f}M"


# --- demo ---
snap = StockSnapshot.from_yfinance("AAPL")
print(f"{snap.ticker}: ${snap.price:.2f} ({snap.change_pct:+.2f}%)")
print(f"Market cap: {snap.market_cap}  P/E: {snap.pe_ratio}")
print(f"52-week range: ${snap.week52_low:.2f} - ${snap.week52_high:.2f}")

Candlestick chart with volume

import plotly.graph_objects as go
from plotly.subplots import make_subplots

ticker = "AAPL"
df = yf.download(ticker, period="6mo", interval="1d", progress=False)

fig = make_subplots(
    rows=2, cols=1, shared_xaxes=True,
    vertical_spacing=0.03,
    row_heights=[0.7, 0.3],
)

# Candlestick
fig.add_trace(go.Candlestick(
    x=df.index,
    open=df["Open"].values.flatten(),
    high=df["High"].values.flatten(),
    low=df["Low"].values.flatten(),
    close=df["Close"].values.flatten(),
    name="Price",
    increasing_line_color="#00CC96",
    decreasing_line_color="#EF553B",
), row=1, col=1)

# Volume bars
colors = [
    "#00CC96" if c >= o else "#EF553B"
    for c, o in zip(
        df["Close"].values.flatten(),
        df["Open"].values.flatten(),
    )
]
fig.add_trace(go.Bar(
    x=df.index,
    y=df["Volume"].values.flatten(),
    marker_color=colors,
    name="Volume",
    showlegend=False,
), row=2, col=1)

fig.update_layout(
    title=f"{ticker} — 6 Month Price & Volume",
    template="plotly_dark",
    xaxis_rangeslider_visible=False,
    margin=dict(t=50, b=40, l=60, r=20),
    showlegend=False,
)
fig.update_yaxes(title_text="Price ($)", row=1, col=1)
fig.update_yaxes(title_text="Volume", row=2, col=1)
fig.show()

52-week range gauge

snap = StockSnapshot.from_yfinance("AAPL")

fig = go.Figure(go.Indicator(
    mode="gauge+number+delta",
    value=snap.price,
    delta={"reference": snap.week52_low, "relative": True, "valueformat": ".1%"},
    title={"text": f"{snap.ticker} vs 52-Week Range"},
    gauge={
        "axis": {"range": [snap.week52_low, snap.week52_high]},
        "bar": {"color": "#636EFA"},
        "steps": [
            {"range": [snap.week52_low, snap.week52_high * 0.5 + snap.week52_low * 0.5], "color": "#EF553B"},
            {"range": [snap.week52_high * 0.5 + snap.week52_low * 0.5, snap.week52_high], "color": "#00CC96"},
        ],
    },
))
fig.update_layout(template="plotly_dark", margin=dict(t=60, b=20, l=30, r=30))
fig.show()

Key takeaways

  • make_subplots with shared_xaxes=True keeps the candlestick and volume chart aligned. Drag-to-zoom on one zooms both.
  • Color-coding volume bars by candle direction (green for up, red for down) shows conviction behind price moves.
  • The gauge chart gives an instant read on where the current price sits within its 52-week range — near the low (potential value) or near the high (potential overextension).
  • Wrap StockSnapshot.from_yfinance in a loop over a watchlist to build a full dashboard table programmatically.
