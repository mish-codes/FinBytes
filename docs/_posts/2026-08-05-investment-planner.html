Compound growth with contributions

The future value of a lump sum plus regular monthly contributions is:

FV = P * (1 + r)^n  +  C * ((1 + r)^n - 1) / r

P = initial principal
C = monthly contribution
r = monthly rate (annual_rate / 12)
n = total months

The first term is growth on the initial deposit. The second is the annuity — growth on the stream of contributions.

Planner code

from dataclasses import dataclass


@dataclass
class GrowthRow:
    month: int
    contribution_total: float
    growth: float
    balance: float


def investment_growth(
    principal: float,
    monthly_contribution: float,
    annual_rate: float,
    years: int,
) -> list[GrowthRow]:
    """Month-by-month investment growth schedule.

    Parameters
    ----------
    principal : float
        Starting investment.
    monthly_contribution : float
        Amount added each month.
    annual_rate : float
        Expected annual return as a decimal (e.g. 0.08).
    years : int
        Investment horizon.

    Returns
    -------
    list[GrowthRow]
    """
    r = annual_rate / 12
    balance = principal
    contributions = principal
    schedule: list[GrowthRow] = []

    for month in range(1, years * 12 + 1):
        balance = balance * (1 + r) + monthly_contribution
        contributions += monthly_contribution
        growth = round(balance - contributions, 2)
        schedule.append(GrowthRow(
            month=month,
            contribution_total=round(contributions, 2),
            growth=growth,
            balance=round(balance, 2),
        ))

    return schedule


def what_if(
    principal: float,
    monthly: float,
    years: int,
    rates: list[float],
) -> dict[str, float]:
    """Quick what-if: final balance for each annual return."""
    result: dict[str, float] = {}
    for rate in rates:
        rows = investment_growth(principal, monthly, rate, years)
        result[f"{rate:.1%}"] = rows[-1].balance
    return result


# --- demo ---
rows = investment_growth(
    principal=10_000,
    monthly_contribution=500,
    annual_rate=0.08,
    years=20,
)

final = rows[-1]
print(f"Total contributed : ${final.contribution_total:,.2f}")
print(f"Investment growth : ${final.growth:,.2f}")
print(f"Final balance     : ${final.balance:,.2f}")

print("\nWhat-if scenarios (same inputs, different returns):")
for label, bal in what_if(10_000, 500, 20, [0.05, 0.08, 0.10]).items():
    print(f"  {label} -> ${bal:,.0f}")

Contributions vs growth stacked chart

import plotly.graph_objects as go

rows = investment_growth(10_000, 500, 0.08, 20)

months = [r.month for r in rows]
contribs = [r.contribution_total for r in rows]
growths = [r.growth for r in rows]

fig = go.Figure()
fig.add_trace(go.Scatter(
    x=months, y=contribs,
    mode="lines", name="Total contributions",
    stackgroup="one",
    line=dict(color="#636EFA"),
))
fig.add_trace(go.Scatter(
    x=months, y=growths,
    mode="lines", name="Investment growth",
    stackgroup="one",
    line=dict(color="#00CC96"),
))
fig.update_layout(
    title="$10 K + $500/mo at 8 % Annual Return",
    xaxis_title="Month",
    yaxis_title="Dollars ($)",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
)
fig.show()

Key takeaways

  • At 8 % over 20 years, growth exceeds contributions — compounding does most of the heavy lifting in the second half.
  • The gap between 5 % and 10 % return is $176,000 on the same contributions, showing how sensitive outcomes are to return assumptions.
  • Use the what_if helper to stress-test different market environments before committing to a plan.
