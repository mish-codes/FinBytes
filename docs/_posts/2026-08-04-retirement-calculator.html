The idea

A deterministic projection uses one assumed return. Reality is messier — some years the market drops 20 %, others it rallies 30 %. A Monte Carlo simulation runs thousands of random return paths to show the range of possible outcomes, giving you percentile bands (p10, p50, p90) instead of a single line.

Core calculator

import numpy as np
from dataclasses import dataclass


@dataclass
class RetirementResult:
    """Holds Monte Carlo simulation output."""
    years: np.ndarray           # shape (horizon,)
    p10: np.ndarray             # 10th percentile path
    p50: np.ndarray             # median path
    p90: np.ndarray             # 90th percentile path
    prob_success: float         # fraction of paths that hit the goal


def retirement_monte_carlo(
    current_savings: float,
    annual_contribution: float,
    expected_return: float,
    volatility: float,
    horizon_years: int,
    goal: float,
    n_simulations: int = 5_000,
    seed: int = 42,
) -> RetirementResult:
    """Run a Monte Carlo projection of retirement savings.

    Parameters
    ----------
    current_savings : float
        Starting portfolio value.
    annual_contribution : float
        Amount added at the end of each year.
    expected_return : float
        Mean annual return (e.g. 0.07 for 7 %).
    volatility : float
        Annual standard deviation of returns (e.g. 0.15).
    horizon_years : int
        Number of years to project.
    goal : float
        Target nest egg at the end of the horizon.
    n_simulations : int
        Number of random paths to simulate.
    seed : int
        Random seed for reproducibility.

    Returns
    -------
    RetirementResult
    """
    rng = np.random.default_rng(seed)

    # Annual log-normal returns: shape (n_simulations, horizon_years)
    annual_returns = rng.normal(
        loc=expected_return - 0.5 * volatility ** 2,
        scale=volatility,
        size=(n_simulations, horizon_years),
    )

    balances = np.zeros((n_simulations, horizon_years))

    for t in range(horizon_years):
        prev = balances[:, t - 1] if t > 0 else current_savings
        balances[:, t] = (prev + annual_contribution) * np.exp(annual_returns[:, t])

    years = np.arange(1, horizon_years + 1)
    p10 = np.percentile(balances, 10, axis=0)
    p50 = np.percentile(balances, 50, axis=0)
    p90 = np.percentile(balances, 90, axis=0)
    prob_success = float(np.mean(balances[:, -1] >= goal))

    return RetirementResult(years, p10, p50, p90, prob_success)


# --- demo ---
result = retirement_monte_carlo(
    current_savings=50_000,
    annual_contribution=20_000,
    expected_return=0.07,
    volatility=0.15,
    horizon_years=30,
    goal=1_500_000,
)

print(f"Median final balance : ${result.p50[-1]:,.0f}")
print(f"10th percentile      : ${result.p10[-1]:,.0f}")
print(f"90th percentile      : ${result.p90[-1]:,.0f}")
print(f"P(reach $1.5 M)      : {result.prob_success:.1%}")

Monte Carlo fan chart

import plotly.graph_objects as go

result = retirement_monte_carlo(
    current_savings=50_000,
    annual_contribution=20_000,
    expected_return=0.07,
    volatility=0.15,
    horizon_years=30,
    goal=1_500_000,
)

fig = go.Figure()

# p10-p90 band
fig.add_trace(go.Scatter(
    x=list(result.years) + list(result.years[::-1]),
    y=list(result.p90) + list(result.p10[::-1]),
    fill="toself",
    fillcolor="rgba(99,110,250,0.2)",
    line=dict(color="rgba(0,0,0,0)"),
    name="p10 - p90 range",
))

# median line
fig.add_trace(go.Scatter(
    x=result.years, y=result.p50,
    mode="lines", name="Median (p50)",
    line=dict(color="#636EFA", width=2),
))

# goal threshold
fig.add_hline(
    y=1_500_000, line_dash="dash",
    line_color="#EF553B",
    annotation_text="Goal: $1.5 M",
    annotation_position="top left",
)

fig.update_layout(
    title=f"Retirement Projection — {result.prob_success:.0%} chance of reaching goal",
    xaxis_title="Year",
    yaxis_title="Portfolio value ($)",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
)
fig.show()

Key takeaways

  • The fan chart shows uncertainty grows over time — a single projected line hides enormous variance.
  • Starting early matters: the same $20 K/yr over 30 years dominates $40 K/yr over 15 years because of compounding.
  • Adjust volatility to model conservative (bonds-heavy) vs aggressive (equities-heavy) portfolios.
