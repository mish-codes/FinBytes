The approach

A currency dashboard needs three things: live exchange rates, a conversion function, and a historical chart. The code below fetches spot rates from a free API, builds a rate matrix, and plots historical forex data using yfinance.

Fetch live exchange rates

import requests
from dataclasses import dataclass


@dataclass
class CurrencyConverter:
    """Convert between currencies using live rates."""
    base: str = "USD"
    rates: dict[str, float] | None = None

    def fetch_rates(self) -> None:
        """Fetch rates from exchangerate.host (free, no key required)."""
        url = f"https://api.exchangerate.host/latest?base={self.base}"
        resp = requests.get(url, timeout=10)
        resp.raise_for_status()
        self.rates = resp.json()["rates"]

    def convert(self, amount: float, src: str, dst: str) -> float:
        if self.rates is None:
            self.fetch_rates()
        # Convert src -> base -> dst
        in_base = amount / self.rates[src]
        return round(in_base * self.rates[dst], 4)

    def rate(self, src: str, dst: str) -> float:
        return self.convert(1.0, src, dst)


# --- demo ---
fx = CurrencyConverter(base="USD")
fx.fetch_rates()

pairs = [("USD", "EUR"), ("USD", "GBP"), ("EUR", "JPY"), ("GBP", "CHF")]
for src, dst in pairs:
    print(f"1 {src} = {fx.rate(src, dst):.4f} {dst}")

Rate matrix

A cross-rate matrix shows every pair at a glance. Each cell is the price of one unit of the row currency in terms of the column currency.

import pandas as pd

currencies = ["USD", "EUR", "GBP", "JPY", "CHF", "CAD", "AUD"]

matrix = pd.DataFrame(index=currencies, columns=currencies, dtype=float)
for src in currencies:
    for dst in currencies:
        matrix.loc[src, dst] = fx.rate(src, dst)

print(matrix.to_string(float_format=lambda x: f"{x:.4f}"))

Historical forex chart

import yfinance as yf
import plotly.graph_objects as go

# yfinance forex tickers use the format "EURUSD=X"
pair = "EURUSD=X"
df = yf.download(pair, period="1y", interval="1d", progress=False)

fig = go.Figure()
fig.add_trace(go.Scatter(
    x=df.index, y=df["Close"].values.flatten(),
    mode="lines",
    name="EUR/USD",
    line=dict(color="#636EFA", width=2),
))
fig.add_trace(go.Scatter(
    x=df.index,
    y=df["Close"].rolling(50).mean().values.flatten(),
    mode="lines",
    name="50-day MA",
    line=dict(color="#EF553B", width=1, dash="dash"),
))
fig.update_layout(
    title="EUR/USD — 1 Year",
    xaxis_title="Date",
    yaxis_title="Exchange Rate",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
    hovermode="x unified",
)
fig.show()

Multi-pair comparison

pairs = {"EURUSD=X": "EUR/USD", "GBPUSD=X": "GBP/USD", "USDJPY=X": "USD/JPY"}

fig = go.Figure()
for ticker, label in pairs.items():
    df = yf.download(ticker, period="6mo", interval="1d", progress=False)
    # Normalize to percentage change from first day
    close = df["Close"].values.flatten()
    pct = (close / close[0] - 1) * 100
    fig.add_trace(go.Scatter(
        x=df.index, y=pct, mode="lines", name=label,
    ))

fig.update_layout(
    title="Forex Pair Performance (6 months, % change)",
    yaxis_title="% Change",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
    hovermode="x unified",
)
fig.show()

Key takeaways

  • exchangerate.host provides free spot rates with no API key. For production, add caching so you don't hit rate limits.
  • yfinance forex tickers follow the pattern EURUSD=X — six-letter pair plus =X.
  • Normalizing to percentage change lets you compare pairs with very different scales (EUR/USD ~1.08 vs USD/JPY ~150) on the same chart.
  • The cross-rate matrix is useful for spotting arbitrage — if A→B→C→A yields more than 1.0, something is mispriced.
