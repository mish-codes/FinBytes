The approach

A financial report pulls stock data, computes summary statistics, formats a metrics table, generates a chart, and exports to CSV — all programmatically. The code below builds a complete report for any list of tickers using yfinance.

Fetch and compute summary statistics

import yfinance as yf
import pandas as pd
import numpy as np
from dataclasses import dataclass


@dataclass
class StockReport:
    """Summary statistics for a single ticker over a given period."""
    ticker: str
    start_price: float
    end_price: float
    high: float
    low: float
    total_return_pct: float
    annualized_vol: float
    avg_daily_volume: int
    sharpe_ratio: float

    @classmethod
    def from_ticker(cls, ticker: str, period: str = "1y") -> "StockReport":
        df = yf.download(ticker, period=period, interval="1d", progress=False)
        close = df["Close"].values.flatten()
        volume = df["Volume"].values.flatten()

        daily_returns = np.diff(close) / close[:-1]
        total_return = (close[-1] / close[0] - 1) * 100
        ann_vol = np.std(daily_returns) * np.sqrt(252) * 100
        sharpe = (np.mean(daily_returns) * 252) / (np.std(daily_returns) * np.sqrt(252))

        return cls(
            ticker=ticker,
            start_price=round(close[0], 2),
            end_price=round(close[-1], 2),
            high=round(np.max(close), 2),
            low=round(np.min(close), 2),
            total_return_pct=round(total_return, 2),
            annualized_vol=round(ann_vol, 2),
            avg_daily_volume=int(np.mean(volume)),
            sharpe_ratio=round(sharpe, 2),
        )


# --- demo ---
report = StockReport.from_ticker("AAPL")
print(f"{report.ticker}: {report.total_return_pct:+.2f}% return, "
      f"{report.annualized_vol:.1f}% vol, Sharpe {report.sharpe_ratio:.2f}")

Multi-stock report table

tickers = ["AAPL", "MSFT", "GOOGL", "AMZN", "NVDA"]
reports = [StockReport.from_ticker(t) for t in tickers]

report_df = pd.DataFrame([

        "Sharpe": f"{r.sharpe_ratio:.2f}",

    for r in reports

Performance comparison chart

import plotly.graph_objects as go

fig = go.Figure()
for r in reports:
    df = yf.download(r.ticker, period="1y", interval="1d", progress=False)
    close = df["Close"].values.flatten()
    pct = (close / close[0] - 1) * 100
    fig.add_trace(go.Scatter(
        x=df.index, y=pct, mode="lines", name=r.ticker,
    ))

fig.update_layout(
    title="1-Year Normalized Returns (%)",
    yaxis_title="% Return",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
    hovermode="x unified",
)
fig.show()

Export to CSV

def export_report(reports: list[StockReport], filename: str = "stock_report.csv") -> str:
    """Export report data to a downloadable CSV."""
    rows = []
    for r in reports:
        rows.append({
            "ticker": r.ticker,
            "start_price": r.start_price,
            "end_price": r.end_price,
            "high": r.high,
            "low": r.low,
            "total_return_pct": r.total_return_pct,
            "annualized_vol_pct": r.annualized_vol,
            "sharpe_ratio": r.sharpe_ratio,
            "avg_daily_volume": r.avg_daily_volume,
        })
    df = pd.DataFrame(rows)
    df.to_csv(filename, index=False)
    return filename


path = export_report(reports)
print(f"Report saved to {path}")

The CSV output is ready for Excel, Google Sheets, or further analysis. For a scheduled report, wrap the entire pipeline in a function and call it with cron or a task scheduler.

Key takeaways

  • The Sharpe ratio normalizes return by risk. A Sharpe above 1.0 means the return more than compensates for the volatility taken.
  • The risk-return scatter instantly shows which stocks gave the most return per unit of risk. Bubble size encodes Sharpe so the best risk-adjusted performers jump out visually.
  • Normalizing prices to percentage change (starting at 0%) lets you compare stocks with very different price levels on the same chart.
  • Automate this with a ticker list and a cron job: python report.py --tickers AAPL MSFT GOOGL --period 1y --output weekly_report.csv.
