Why compare?

A lower rate does not always mean the cheapest loan. Shorter terms raise the monthly payment but slash total interest. This tool puts two or three loan options side by side so you can see the trade-off at a glance.

Comparison engine

from dataclasses import dataclass


@dataclass
class LoanSummary:
    label: str
    principal: float
    annual_rate: float
    years: int
    monthly_payment: float
    total_interest: float
    total_cost: float


def pmt(principal: float, annual_rate: float, years: int) -> float:
    r = annual_rate / 12
    n = years * 12
    if r == 0:
        return round(principal / n, 2)
    return round(principal * r * (1 + r) ** n / ((1 + r) ** n - 1), 2)


def summarise_loan(
    label: str,
    principal: float,
    annual_rate: float,
    years: int,
) -> LoanSummary:
    """Compute headline numbers for a single loan."""
    monthly = pmt(principal, annual_rate, years)
    n = years * 12
    total_cost = round(monthly * n, 2)
    total_interest = round(total_cost - principal, 2)
    return LoanSummary(
        label=label,
        principal=principal,
        annual_rate=annual_rate,
        years=years,
        monthly_payment=monthly,
        total_interest=total_interest,
        total_cost=total_cost,
    )


def compare_loans(
    principal: float,
    options: list[tuple[str, float, int]],
) -> list[LoanSummary]:
    """Compare multiple (label, rate, years) options.

    Parameters
    ----------
    principal : float
        Loan amount (same for all options).
    options : list[tuple[str, float, int]]
        Each tuple is (label, annual_rate, term_years).

    Returns
    -------
    list[LoanSummary]
    """
    return [
        summarise_loan(label, principal, rate, years)
        for label, rate, years in options
    ]


# --- demo ---
loans = compare_loans(
    principal=300_000,
    options=[
        ("30-yr @ 6.5 %", 0.065, 30),
        ("20-yr @ 6.0 %", 0.060, 20),
        ("15-yr @ 5.5 %", 0.055, 15),
    ],
)

for ln in loans:
    print(f"{ln.label:18s}  PMT ${ln.monthly_payment:>9,.2f}  "
          f"Interest ${ln.total_interest:>11,.2f}  "
          f"Total ${ln.total_cost:>11,.2f}")

Total cost bar chart

import plotly.graph_objects as go

loans = compare_loans(
    principal=300_000,
    options=[
        ("30-yr @ 6.5 %", 0.065, 30),
        ("20-yr @ 6.0 %", 0.060, 20),
        ("15-yr @ 5.5 %", 0.055, 15),
    ],
)

labels = [ln.label for ln in loans]
principals = [ln.principal for ln in loans]
interests = [ln.total_interest for ln in loans]

fig = go.Figure()
fig.add_trace(go.Bar(
    x=labels, y=principals,
    name="Principal",
    marker_color="#636EFA",
))
fig.add_trace(go.Bar(
    x=labels, y=interests,
    name="Interest",
    marker_color="#EF553B",
))
fig.update_layout(
    barmode="stack",
    title="Total Loan Cost: Principal + Interest",
    yaxis_title="Dollars ($)",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
)
fig.show()

Key takeaways

  • Switching from 30 to 15 years saves over $241,000 in interest, but your monthly payment increases by about $555.
  • The 20-year option is a middle ground — meaningful interest savings with a more manageable payment bump.
  • Always compare the total cost, not just the monthly payment, to see the true price of a loan.
