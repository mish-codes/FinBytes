What we are building

Two classic systematic strategies, backtested on synthetic price data: a SMA crossover (buy when the fast moving average crosses above the slow) and a momentum strategy (buy when the N-day return is positive). We will compare equity curves, Sharpe ratios, and trade logs.

Synthetic price data

import numpy as np
import pandas as pd

np.random.seed(42)

n_days = 504  # ~2 years
dates = pd.date_range("2024-01-01", periods=n_days, freq="B")

# Trending random walk with mean reversion
returns = np.random.normal(0.0003, 0.013, n_days)
# Add a mild trend regime
returns[100:200] += 0.001  # bull run
returns[300:350] -= 0.002  # correction

prices = 100 * np.cumprod(1 + returns)
df = pd.DataFrame({"date": dates, "price": prices})
df = df.set_index("date")

print(f"Start: ${df['price'].iloc[0]:.2f}  End: ${df['price'].iloc[-1]:.2f}")
print(f"Buy & hold return: {(df['price'].iloc[-1] / df['price'].iloc[0] - 1):.2%}")

SMA crossover strategy

def sma_crossover(prices: pd.Series, fast: int = 20, slow: int = 50) -> pd.DataFrame:
    """Generate signals: 1 = long, 0 = flat."""
    df = pd.DataFrame({"price": prices})
    df["sma_fast"] = prices.rolling(fast).mean()
    df["sma_slow"] = prices.rolling(slow).mean()

    df["signal"] = 0
    df.loc[df["sma_fast"] > df["sma_slow"], "signal"] = 1

    # Detect crossover points
    df["position_change"] = df["signal"].diff()
    df["trade"] = "hold"
    df.loc[df["position_change"] == 1, "trade"] = "buy"
    df.loc[df["position_change"] == -1, "trade"] = "sell"

    return df

sma_df = sma_crossover(df["price"], fast=20, slow=50)
n_trades = (sma_df["trade"] != "hold").sum()
print(f"SMA Crossover: {n_trades} trades")

Momentum strategy

def momentum_strategy(prices: pd.Series, lookback: int = 20) -> pd.DataFrame:
    """Go long when N-day return is positive."""
    df = pd.DataFrame({"price": prices})
    df["momentum"] = prices.pct_change(lookback)

    df["signal"] = 0
    df.loc[df["momentum"] > 0, "signal"] = 1

    df["position_change"] = df["signal"].diff()
    df["trade"] = "hold"
    df.loc[df["position_change"] == 1, "trade"] = "buy"
    df.loc[df["position_change"] == -1, "trade"] = "sell"

    return df

mom_df = momentum_strategy(df["price"], lookback=20)
n_trades_mom = (mom_df["trade"] != "hold").sum()
print(f"Momentum: {n_trades_mom} trades")

Backtest engine

def backtest(prices: pd.Series, signals: pd.Series) -> pd.DataFrame:
    """Compute equity curve from price series and signal series (1=long, 0=flat)."""
    daily_returns = prices.pct_change().fillna(0)
    strategy_returns = daily_returns * signals.shift(1).fillna(0)

    result = pd.DataFrame({
        "price": prices,
        "signal": signals,
        "daily_return": daily_returns,
        "strategy_return": strategy_returns,
    })
    result["equity"] = (1 + result["strategy_return"]).cumprod()
    result["buy_hold"] = prices / prices.iloc[0]

    return result

sma_bt = backtest(df["price"], sma_df["signal"])
mom_bt = backtest(df["price"], mom_df["signal"])

Performance metrics

def calc_metrics(bt: pd.DataFrame, name: str) -> dict:
    """Compute key backtest statistics."""
    total_return = bt["equity"].iloc[-1] - 1
    n_years = len(bt) / 252
    annual_return = (1 + total_return) ** (1 / n_years) - 1

    daily_std = bt["strategy_return"].std()
    sharpe = bt["strategy_return"].mean() / daily_std * np.sqrt(252) if daily_std > 0 else 0

    rolling_max = bt["equity"].cummax()
    drawdown = (bt["equity"] - rolling_max) / rolling_max
    max_drawdown = drawdown.min()

    pct_in_market = bt["signal"].mean()

    return {
        "Strategy": name,
        "Total Return": f"{total_return:.2%}",
        "Annual Return": f"{annual_return:.2%}",
        "Sharpe Ratio": f"{sharpe:.3f}",
        "Max Drawdown": f"{max_drawdown:.2%}",
        "% Time in Market": f"{pct_in_market:.0%}",
    }

bh_return = df["price"].iloc[-1] / df["price"].iloc[0] - 1
metrics = pd.DataFrame([
    calc_metrics(sma_bt, "SMA Crossover"),
    calc_metrics(mom_bt, "Momentum"),
    {"Strategy": "Buy & Hold", "Total Return": f"{bh_return:.2%}",
     "Annual Return": f"{((1+bh_return)**(1/(len(df)/252))-1):.2%}",
     "Sharpe Ratio": f"{df['price'].pct_change().mean()/df['price'].pct_change().std()*np.sqrt(252):.3f}",
     "Max Drawdown": f"{((df['price']/df['price'].cummax()-1).min()):.2%}",
     "% Time in Market": "100%"},
])
print(metrics.to_string(index=False))

Trade log

def trade_log(strategy_df: pd.DataFrame) -> pd.DataFrame:
    """Extract a log of all buy/sell trades with P&L."""
    trades = strategy_df[strategy_df["trade"].isin(["buy", "sell"])].copy()
    trades = trades[["price", "trade"]].reset_index()

    log = []
    for i in range(0, len(trades) - 1, 2):
        if trades.iloc[i]["trade"] == "buy" and i + 1 < len(trades):
            entry = trades.iloc[i]
            exit_ = trades.iloc[i + 1]
            pnl_pct = (exit_["price"] / entry["price"] - 1) * 100
            log.append({
                "entry_date": entry["date"],
                "exit_date": exit_["date"],
                "entry_price": round(entry["price"], 2),
                "exit_price": round(exit_["price"], 2),
                "pnl_pct": round(pnl_pct, 2),
            })

    return pd.DataFrame(log)

sma_trades = trade_log(sma_df)
print(f"SMA Crossover Trade Log ({len(sma_trades)} round trips):")
print(sma_trades.to_string(index=False))

Backtest pitfalls

  • Transaction costs — not included here. Each trade costs a spread and commission. High-frequency strategies can be profitable before costs and unprofitable after.
  • Slippage — we assume execution at the closing price. In reality, your fill price will differ.
  • Overfitting — optimizing parameters (fast=20, slow=50) on the same data you test on inflates results. Use walk-forward validation.
  • Survivorship bias — testing on stocks that still exist today ignores the ones that went to zero.
