The task

Predict next-day stock returns using engineered features: lagged returns, moving averages, RSI, and volume changes. We use a Random Forest regressor, evaluate on a proper train/test split, and discuss why this is educational but not a trading system.

Generate synthetic stock data

import numpy as np
import pandas as pd

np.random.seed(42)

n_days = 756  # ~3 years
dates = pd.date_range("2023-01-01", periods=n_days, freq="B")

returns = np.random.normal(0.0003, 0.014, n_days)
# Add slight autocorrelation (momentum)
for i in range(1, n_days):
    returns[i] += 0.05 * returns[i - 1]

prices = 100 * np.cumprod(1 + returns)
volume = np.random.lognormal(mean=15, sigma=0.3, size=n_days)

df = pd.DataFrame({"date": dates, "price": prices, "volume": volume})
df = df.set_index("date")
df["return"] = df["price"].pct_change()

print(f"Data range: {df.index[0].date()} to {df.index[-1].date()}")
print(f"Price range: ${df['price'].min():.2f} - ${df['price'].max():.2f}")

Feature engineering

def compute_rsi(prices: pd.Series, window: int = 14) -> pd.Series:
    """Relative Strength Index."""
    delta = prices.diff()
    gain = delta.where(delta > 0, 0).rolling(window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window).mean()
    rs = gain / loss
    return 100 - (100 / (1 + rs))

def build_features(df: pd.DataFrame) -> pd.DataFrame:
    """Create ML features from price and volume data."""
    feat = pd.DataFrame(index=df.index)

    # Lagged returns
    for lag in [1, 2, 3, 5, 10]:
        feat[f"return_lag{lag}"] = df["return"].shift(lag)

    # Moving averages (ratio to current price)
    for window in [5, 10, 20, 50]:
        sma = df["price"].rolling(window).mean()
        feat[f"sma{window}_ratio"] = df["price"] / sma

    # RSI
    feat["rsi_14"] = compute_rsi(df["price"], 14)

    # Volatility
    feat["vol_10d"] = df["return"].rolling(10).std()
    feat["vol_20d"] = df["return"].rolling(20).std()

    # Volume change
    feat["volume_change"] = df["volume"].pct_change()
    feat["volume_sma_ratio"] = df["volume"] / df["volume"].rolling(20).mean()

    # Target: next-day return
    feat["target"] = df["return"].shift(-1)

    return feat.dropna()

features_df = build_features(df)
print(f"Features: {features_df.shape[1] - 1}")
print(f"Samples:  {features_df.shape[0]}")
print(features_df.columns.tolist())

Every feature uses only past data. shift(lag) looks backward, and the target is shift(-1) (tomorrow's return). This avoids look-ahead bias. The SMA ratio tells the model whether the price is above or below its trend. RSI captures overbought/oversold conditions.

Train/test split

feature_cols = [c for c in features_df.columns if c != "target"]
X = features_df[feature_cols]
y = features_df["target"]

# Time series split: train on first 70%, test on last 30%
split_idx = int(len(X) * 0.7)
X_train, X_test = X.iloc[:split_idx], X.iloc[split_idx:]
y_train, y_test = y.iloc[:split_idx], y.iloc[split_idx:]

print(f"Train: {X_train.shape[0]} ({X_train.index[0].date()} to {X_train.index[-1].date()})")
print(f"Test:  {X_test.shape[0]} ({X_test.index[0].date()} to {X_test.index[-1].date()})")

Never use random train/test split for time series. Future data must not leak into training. We use a chronological split: train on the past, test on the future.

Random Forest model

from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error

model = RandomForestRegressor(
    n_estimators=300,
    max_depth=8,
    min_samples_leaf=10,
    random_state=42,
    n_jobs=-1,
)
model.fit(X_train, y_train)

y_pred = model.predict(X_test)

mae = mean_absolute_error(y_test, y_pred)
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
corr = np.corrcoef(y_test, y_pred)[0, 1]

print(f"MAE:  {mae:.6f}")
print(f"RMSE: {rmse:.6f}")
print(f"Correlation (actual vs predicted): {corr:.4f}")

Directional accuracy

# Can the model at least predict the direction?
direction_correct = np.sign(y_pred) == np.sign(y_test.values)
directional_accuracy = direction_correct.mean()
print(f"Directional accuracy: {directional_accuracy:.1%}")
print(f"Random baseline:      50.0%")

# Sharpe of a strategy that goes long when predicted return > 0
strategy_returns = y_test.values * np.sign(y_pred)
strat_sharpe = strategy_returns.mean() / strategy_returns.std() * np.sqrt(252)
bh_sharpe = y_test.mean() / y_test.std() * np.sqrt(252)
print(f"\nStrategy Sharpe:   {strat_sharpe:.3f}")
print(f"Buy & Hold Sharpe: {bh_sharpe:.3f}")

Reality check

  • Markets are efficient-ish — if simple features predicted returns reliably, everyone would trade on them and the signal would vanish. A directional accuracy above 52% with real data and transaction costs is exceptional.
  • Overfitting risk — Random Forest with many features on noisy data will overfit. Use cross-validation (e.g., TimeSeriesSplit) and limit tree depth.
  • Feature decay — features that work in one regime (bull market) may fail in another (crash). Retrain regularly.
  • This is educational — do not trade real money based on this model. Production systems need tick-level data, execution infrastructure, and rigorous out-of-sample testing.
