Two Tools, One Goal

Python gives you two ways to replace real objects during tests: pytest's built-in monkeypatch fixture and the standard library's unittest.mock.patch. They overlap, but each has a sweet spot. This reference covers both with concrete examples from financial Python code.

Feature monkeypatch unittest.mock.patch
Built into pytest Yes No (stdlib)
Scope Function only Function, class, module
Call tracking No (you wire it yourself) Yes (call_count, call_args, etc.)
side_effect sequences No Yes
Env var support setenv / delenv patch.dict(os.environ, ...)
Works outside pytest No Yes (unittest, plain scripts)

Rule of thumb: reach for monkeypatch when you need a simple replacement with no call inspection. Reach for mock.patch when you need to verify how something was called, or need side_effect sequences.

The "Patch Where It's Used" Rule

This is the single most common mocking mistake. When module A imports a name from module B, you must patch the name in module A's namespace, not in module B.

# market_data.py
def fetch_prices(ticker: str) -> dict:
    """Calls a real API - slow, requires network."""
    import yfinance as yf
    data = yf.download(ticker, period="5d")
    return data["Close"].to_dict()

# scanner.py
from market_data import fetch_prices   # <-- binds name in scanner's namespace

def scan_ticker(ticker: str) -> str:
    prices = fetch_prices(ticker)       # <-- this calls scanner.fetch_prices
    if not prices:
        return "NO_DATA"
    latest = list(prices.values())[-1]
    return "BUY" if latest > 150 else "HOLD"

# test_scanner.py

# WRONG - this patches the original, but scanner already has its own reference:
# @patch("market_data.fetch_prices")

# RIGHT - patch where it's USED:
from unittest.mock import patch

def test_scan_ticker_buy():
    fake_prices = {"2026-07-15": 160.0, "2026-07-16": 165.0}
    with patch("scanner.fetch_prices", return_value=fake_prices):
        assert scan_ticker("AAPL") == "BUY"

def test_scan_ticker_no_data():
    with patch("scanner.fetch_prices", return_value={}):
        assert scan_ticker("AAPL") == "NO_DATA"

Why does this happen? from market_data import fetch_prices copies a reference into scanner's global namespace. Patching the original in market_data doesn't update scanner's copy. Always patch at the call site.

monkeypatch Patterns

Replace a method on an object

def test_portfolio_value(monkeypatch):
    """Replace a method with a lambda that returns a known value."""
    import portfolio

    monkeypatch.setattr(
        portfolio.PriceFeed, "get_price",
        lambda self, ticker: 150.0
    )
    pf = portfolio.Portfolio(["AAPL", "MSFT"], quantities=[10, 20])
    # get_price always returns 150.0, so total = (10 + 20) * 150.0
    assert pf.total_value() == 4500.0

Replace by string path

def test_fetch_treasury_rates(monkeypatch):
    """String path is useful when you don't want to import the module."""
    monkeypatch.setattr(
        "services.fred_client.fetch_series",
        lambda series_id, start, end: [4.25, 4.30, 4.28]
    )
    from services.fred_client import fetch_series
    assert fetch_series("DGS10", "2026-01-01", "2026-01-03") == [4.25, 4.30, 4.28]

Environment variables

def test_api_key_loaded(monkeypatch):
    """Set (or override) an environment variable for one test."""
    monkeypatch.setenv("FRED_API_KEY", "test-key-abc123")

    import os
    assert os.environ["FRED_API_KEY"] == "test-key-abc123"

def test_missing_api_key(monkeypatch):
    """Delete an env var to test the 'not configured' path."""
    monkeypatch.delenv("FRED_API_KEY", raising=False)

    from config import get_api_key
    with pytest.raises(EnvironmentError, match="FRED_API_KEY not set"):
        get_api_key()

Delete an attribute

def test_fallback_when_no_cache(monkeypatch):
    """Test behavior when an optional dependency is absent."""
    import services.cache_layer
    monkeypatch.delattr(services.cache_layer, "redis_client", raising=False)

    # Now cache_layer should fall back to in-memory dict
    result = services.cache_layer.get_or_compute("key", lambda: 42)
    assert result == 42

Inline return pattern (the one-liner)

def test_bond_pricer(monkeypatch):
    """Quick lambda that ignores all args and returns a fixed value."""
    import bond_math
    monkeypatch.setattr(bond_math, "bootstrap_spot_curve",
                        lambda *a, **kw: [0.04, 0.042, 0.045])
    price = bond_math.price_bond(face=1000, coupon_rate=0.05, maturity=3)
    assert price == pytest.approx(1022.35, rel=1e-2)

unittest.mock Patterns

Decorator style

from unittest.mock import patch, MagicMock

@patch("scanner.fetch_prices")
def test_scan_returns_hold(mock_fetch):
    mock_fetch.return_value = {"2026-07-16": 120.0}
    result = scan_ticker("MSFT")
    assert result == "HOLD"
    mock_fetch.assert_called_once_with("MSFT")

Context manager style

def test_scan_handles_timeout():
    import requests
    with patch("scanner.fetch_prices") as mock_fetch:
        mock_fetch.side_effect = requests.Timeout("API timed out")
        with pytest.raises(requests.Timeout):
            scan_ticker("AAPL")

side_effect sequences

@patch("risk_engine.get_spot_rate")
def test_var_uses_multiple_rates(mock_rate):
    """Return a different value each time the mock is called."""
    mock_rate.side_effect = [0.04, 0.042, 0.045, 0.039]

    result = risk_engine.compute_var(portfolio_id="p1", num_scenarios=4)
    assert mock_rate.call_count == 4
    assert result > 0

@patch("risk_engine.get_spot_rate")
def test_var_handles_api_failure(mock_rate):
    """Return values, then raise on the 3rd call."""
    mock_rate.side_effect = [0.04, 0.042, ConnectionError("lost connection")]

    with pytest.raises(ConnectionError):
        risk_engine.compute_var(portfolio_id="p1", num_scenarios=4)

Inspecting all calls

@patch("alert_service.send_email")
def test_breach_alert_emails(mock_send):
    """Verify exactly which emails were sent."""
    alert_service.check_limits(portfolio_id="p1")

    # call_args_list stores every call as a tuple of (args, kwargs)
    calls = mock_send.call_args_list
    assert len(calls) == 2
    assert calls[0].kwargs["recipient"] == "risk@firm.com"
    assert calls[1].kwargs["recipient"] == "manager@firm.com"
    assert "VaR breach" in calls[0].kwargs["subject"]

AsyncMock: Mocking Async Functions

from unittest.mock import AsyncMock, patch
import pytest

# --- The code under test ---
# market_ws.py
async def stream_prices(session, ticker: str) -> float:
    """Fetch the latest price from an async HTTP session."""
    resp = await session.get(f"https://api.example.com/price/{ticker}")
    data = await resp.json()
    return data["price"]

# --- The test ---
@pytest.mark.asyncio
async def test_stream_prices():
    """AsyncMock automatically handles 'await'."""
    mock_resp = AsyncMock()
    mock_resp.json.return_value = {"price": 189.50}

    mock_session = AsyncMock()
    mock_session.get.return_value = mock_resp

    price = await stream_prices(mock_session, "AAPL")
    assert price == 189.50
    mock_session.get.assert_called_once_with(
        "https://api.example.com/price/AAPL"
    )

AsyncMock is a MagicMock subclass. When you await it, it returns return_value. When you call it without await, it returns another AsyncMock. This matches how async code chains in practice.

Edge Cases That Trip People Up

1. Patching a property vs a method

class Portfolio:
    @property
    def total_value(self) -> float:
        return sum(self._holdings.values())

# WRONG: this replaces the property with a plain attribute
# monkeypatch.setattr(portfolio, "total_value", 50000.0)

# RIGHT: patch on the class, replace the property itself
from unittest.mock import PropertyMock

with patch.object(type(portfolio), "total_value",
                  new_callable=PropertyMock, return_value=50000.0):
    assert portfolio.total_value == 50000.0

2. Patching a class (mock the class, not the instance)

# risk.py
from db import SessionLocal

def get_positions():
    db = SessionLocal()          # <-- this creates an instance
    return db.query(Position).all()

# test_risk.py
@patch("risk.SessionLocal")
def test_get_positions(MockSession):
    mock_instance = MockSession.return_value  # the instance that () creates
    mock_instance.query.return_value.all.return_value = [
        Position(ticker="AAPL", qty=100)
    ]
    result = get_positions()
    assert len(result) == 1
    assert result[0].ticker == "AAPL"

3. Mocking datetime.now()

from unittest.mock import patch
from datetime import datetime

# Freeze time for a VaR report date check
@patch("report_generator.datetime")
def test_report_uses_today(mock_dt):
    mock_dt.now.return_value = datetime(2026, 7, 20, 9, 30)
    mock_dt.side_effect = lambda *a, **kw: datetime(*a, **kw)  # keep constructors working

    report = report_generator.generate_daily_var()
    assert report["as_of_date"] == "2026-07-20"

Or use freezegun for a cleaner approach:

from freezegun import freeze_time

@freeze_time("2026-07-20 09:30:00")
def test_report_date():
    report = report_generator.generate_daily_var()
    assert report["as_of_date"] == "2026-07-20"

4. Patching builtins (open)

from unittest.mock import patch, mock_open

def test_read_config():
    config_content = '{"db_host": "localhost", "db_port": 5432}'
    with patch("builtins.open", mock_open(read_data=config_content)):
        from config_loader import load_config
        cfg = load_config("config.json")
        assert cfg["db_host"] == "localhost"

5. Scope: monkeypatch is function-scoped only

# This DOES NOT WORK:
# @pytest.fixture(scope="session")
# def patched_api(monkeypatch):  <-- monkeypatch is function-scope only
#     monkeypatch.setattr(...)

# For broader scopes, use mock.patch as a fixture:
@pytest.fixture(scope="module")
def mock_api():
    with patch("scanner.fetch_prices", return_value={"2026-07-16": 155.0}):
        yield

Decision Table: When to Use What

Situation Recommended Tool Why
Replace a function with a fixed return value monkeypatch.setattr Simplest, no imports needed
Set/delete environment variables monkeypatch.setenv Built-in, auto-restores
Verify call count or arguments mock.patch Built-in call tracking
Return different values on successive calls mock.patch + side_effect Not possible with plain lambda
Mock an async function AsyncMock Handles await transparently
Module or session-scoped patch mock.patch in a fixture monkeypatch is function-scope only
Patch a property patch.object + PropertyMock Only reliable way
Quick fixture for a replaced dependency monkeypatch inside @pytest.fixture Clean, pytest-native

Complete Example: Mocking a QuantLab Scanner

Putting it all together. A scanner that fetches prices, checks a threshold, and sends an alert.

# scanner_service.py
from market_data import fetch_prices
from alert_service import send_alert
import os

def run_scan(ticker: str) -> dict:
    threshold = float(os.environ.get("ALERT_THRESHOLD", "200"))
    prices = fetch_prices(ticker)
    latest = list(prices.values())[-1]
    signal = "BUY" if latest > threshold else "HOLD"
    if signal == "BUY":
        send_alert(f"{ticker} above {threshold}: {latest}")
    return {"ticker": ticker, "signal": signal, "price": latest}

# test_scanner_service.py
from unittest.mock import patch, call
import pytest

@patch("scanner_service.send_alert")
@patch("scanner_service.fetch_prices")
def test_run_scan_buy_triggers_alert(mock_fetch, mock_alert, monkeypatch):
    monkeypatch.setenv("ALERT_THRESHOLD", "150")
    mock_fetch.return_value = {"2026-07-16": 175.0}

    result = run_scan("AAPL")

    assert result["signal"] == "BUY"
    assert result["price"] == 175.0
    mock_fetch.assert_called_once_with("AAPL")
    mock_alert.assert_called_once()
    assert "175.0" in mock_alert.call_args[0][0]

@patch("scanner_service.send_alert")
@patch("scanner_service.fetch_prices")
def test_run_scan_hold_no_alert(mock_fetch, mock_alert, monkeypatch):
    monkeypatch.setenv("ALERT_THRESHOLD", "200")
    mock_fetch.return_value = {"2026-07-16": 150.0}

    result = run_scan("MSFT")

    assert result["signal"] == "HOLD"
    mock_alert.assert_not_called()

Notice both tools working together: monkeypatch for the env var (simple replacement), mock.patch for the functions (need call inspection). That's the pattern.
