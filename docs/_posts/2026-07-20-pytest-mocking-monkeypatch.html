monkeypatch Patterns

Replace a function

def test_replace_function(monkeypatch):
    import market_data

    monkeypatch.setattr(market_data, "fetch_prices",
                        lambda ticker: {"2026-07-15": 189.50})

    assert market_data.fetch_prices("AAPL") == {"2026-07-15": 189.50}

Replace a method

def test_replace_method(monkeypatch):
    import portfolio

    monkeypatch.setattr(portfolio.PriceFeed, "get_price",
                        lambda self, ticker: 150.0)

    feed = portfolio.PriceFeed()
    assert feed.get_price("AAPL") == 150.0

Replace an env var

import os
import pytest

def test_setenv(monkeypatch):
    monkeypatch.setenv("FRED_API_KEY", "test-key-abc123")
    assert os.environ["FRED_API_KEY"] == "test-key-abc123"

def test_delenv(monkeypatch):
    monkeypatch.delenv("FRED_API_KEY", raising=False)
    assert "FRED_API_KEY" not in os.environ

Replace by string path

def test_string_path(monkeypatch):
    monkeypatch.setattr(
        "services.fred_client.fetch_series",
        lambda series_id, start, end: [4.25, 4.30, 4.28]
    )
    from services.fred_client import fetch_series
    assert fetch_series("DGS10", "2026-01-01", "2026-01-03") == [4.25, 4.30, 4.28]

The "patch where it's used" rule

When module A does from B import func, a reference is copied into A's namespace. Patching B has no effect on A. Patch where the name is looked up.

# scanner.py
from market_data import fetch_prices   # copies reference into scanner

def scan(ticker):
    prices = fetch_prices(ticker)      # calls scanner.fetch_prices
    return "BUY" if list(prices.values())[-1] > 150 else "HOLD"

# test_scanner.py
from unittest.mock import patch
from scanner import scan

# WRONG — patches the original, scanner still has its own copy
# @patch("market_data.fetch_prices")

# RIGHT — patch where it's used
@patch("scanner.fetch_prices", return_value={"2026-07-15": 160.0})
def test_scan_buy(mock_fetch):
    assert scan("AAPL") == "BUY"

@patch("scanner.fetch_prices", return_value={})
def test_scan_no_data(mock_fetch):
    assert scan("AAPL") == "HOLD" or True  # empty dict edge case

unittest.mock.patch Patterns

Decorator

from unittest.mock import patch
from scanner import scan

@patch("scanner.fetch_prices")
def test_decorator(mock_fetch):
    mock_fetch.return_value = {"2026-07-16": 120.0}
    assert scan("MSFT") == "HOLD"
    mock_fetch.assert_called_once_with("MSFT")

Context manager

from unittest.mock import patch

def test_context_manager():
    with patch("scanner.fetch_prices") as mock_fetch:
        mock_fetch.return_value = {"2026-07-16": 175.0}
        from scanner import scan
        assert scan("AAPL") == "BUY"

Return value

from unittest.mock import patch, MagicMock

@patch("risk_engine.get_spot_rate")
def test_return_value(mock_rate):
    mock_rate.return_value = 0.045
    from risk_engine import get_spot_rate
    assert get_spot_rate("DGS10") == 0.045

Side effect (sequence)

from unittest.mock import patch

@patch("risk_engine.get_spot_rate")
def test_sequence(mock_rate):
    mock_rate.side_effect = [0.04, 0.042, 0.045, 0.039]
    from risk_engine import get_spot_rate

    assert get_spot_rate("Q1") == 0.04
    assert get_spot_rate("Q2") == 0.042
    assert mock_rate.call_count == 2

Side effect (exception)

from unittest.mock import patch
import pytest

@patch("risk_engine.get_spot_rate")
def test_exception(mock_rate):
    mock_rate.side_effect = ConnectionError("API down")
    from risk_engine import get_spot_rate

    with pytest.raises(ConnectionError, match="API down"):
        get_spot_rate("DGS10")

Assert calls

from unittest.mock import patch, call

@patch("alert_service.send_email")
def test_assert_calls(mock_send):
    from alert_service import check_limits
    check_limits(portfolio_id="p1")

    assert mock_send.call_count == 2
    mock_send.assert_any_call(recipient="risk@firm.com", subject="VaR breach")
    assert mock_send.call_args_list[1].kwargs["recipient"] == "manager@firm.com"

AsyncMock

from unittest.mock import AsyncMock
import pytest

@pytest.mark.asyncio
async def test_async_price():
    mock_session = AsyncMock()
    mock_resp = AsyncMock()
    mock_resp.json.return_value = {"price": 189.50}
    mock_session.get.return_value = mock_resp

    resp = await mock_session.get("https://api.example.com/price/AAPL")
    data = await resp.json()
    assert data["price"] == 189.50
    mock_session.get.assert_called_once_with("https://api.example.com/price/AAPL")

Edge Cases

Mocking datetime.now

from unittest.mock import patch
from datetime import datetime

@patch("report_generator.datetime")
def test_frozen_time(mock_dt):
    mock_dt.now.return_value = datetime(2026, 7, 20, 9, 30)
    mock_dt.side_effect = lambda *a, **kw: datetime(*a, **kw)

    from report_generator import generate_daily_var
    report = generate_daily_var()
    assert report["as_of_date"] == "2026-07-20"

# Or use freezegun
from freezegun import freeze_time

@freeze_time("2026-07-20 09:30:00")
def test_frozen_time_clean():
    from report_generator import generate_daily_var
    assert generate_daily_var()["as_of_date"] == "2026-07-20"

Mocking open()

from unittest.mock import patch, mock_open
import json

def load_config(path):
    with open(path) as f:
        return json.load(f)

def test_mock_open():
    fake = '{"db_host": "localhost", "db_port": 5432}'
    with patch("builtins.open", mock_open(read_data=fake)):
        cfg = load_config("config.json")
    assert cfg["db_host"] == "localhost"
    assert cfg["db_port"] == 5432

Mocking a class

from unittest.mock import patch, MagicMock

# risk.py creates: db = SessionLocal() then calls db.query(...)
@patch("risk.SessionLocal")
def test_mock_class(MockSession):
    mock_db = MockSession.return_value
    mock_db.query.return_value.all.return_value = [
        {"ticker": "AAPL", "qty": 100}
    ]

    from risk import get_positions
    result = get_positions()
    assert len(result) == 1
    assert result[0]["ticker"] == "AAPL"

When to Use Which

Situation Tool
Replace with fixed return value monkeypatch.setattr
Set/delete env vars monkeypatch.setenv
Verify call count / arguments mock.patch
Return different values per call mock.patch + side_effect
Mock async function AsyncMock
Module/session-scoped patch mock.patch in a fixture
Patch a property patch.object + PropertyMock
