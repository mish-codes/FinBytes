Basics

Positional argument

import argparse

parser = argparse.ArgumentParser(description="Fetch historical prices")
parser.add_argument("ticker", help="Stock ticker symbol")
args = parser.parse_args()
print(f"Fetching prices for {args.ticker}")

Optional argument

import argparse

parser = argparse.ArgumentParser(description="Bond pricer")
parser.add_argument("bond_id", help="CUSIP or ISIN")
parser.add_argument("--spread", type=float, default=0.0,
                    help="Spread in bps (default: 0)")
parser.add_argument("--curve-date", default="today",
                    help="Yield curve date (default: today)")
args = parser.parse_args()
print(f"Pricing {args.bond_id}, spread={args.spread}bps, date={args.curve_date}")

Choices

import argparse

parser = argparse.ArgumentParser()
parser.add_argument("--side", choices=["buy", "sell"], required=True)
parser.add_argument("--env", choices=["dev", "staging", "prod"], default="dev")
parser.add_argument("--output", "-o", choices=["json", "csv", "table"],
                    default="table")
args = parser.parse_args()
print(f"side={args.side}, env={args.env}, output={args.output}")

Flags

Boolean flag (store_true)

import argparse

parser = argparse.ArgumentParser()
parser.add_argument("--verbose", "-v", action="store_true",
                    help="Enable verbose output")
parser.add_argument("--no-cache", action="store_true",
                    help="Bypass cache")
args = parser.parse_args()
# python script.py --verbose    -> args.verbose = True
# python script.py              -> args.verbose = False
print(f"verbose={args.verbose}, no_cache={args.no_cache}")

The type=bool trap

import argparse

# BROKEN: bool("False") is True because "False" is a non-empty string
parser = argparse.ArgumentParser()
parser.add_argument("--debug", type=bool, default=False)
args = parser.parse_args(["--debug", "False"])
print(f"debug={args.debug}")  # True (!)

# FIX: use store_true
parser2 = argparse.ArgumentParser()
parser2.add_argument("--debug", action="store_true")
args2 = parser2.parse_args([])
print(f"debug={args2.debug}")  # False
args3 = parser2.parse_args(["--debug"])
print(f"debug={args3.debug}")  # True

Subcommands

Full example

"""rds.py -- Manage a QuantLab RDS instance."""
import argparse
import subprocess


def start(args):
    print(f"Starting RDS instance: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "start-db-instance",
        "--db-instance-identifier", args.instance_id,
        "--profile", args.profile,
    ])


def stop(args):
    print(f"Stopping RDS instance: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "stop-db-instance",
        "--db-instance-identifier", args.instance_id,
        "--profile", args.profile,
    ])


def status(args):
    print(f"Checking status: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "describe-db-instances",
        "--db-instance-identifier", args.instance_id,
        "--query", "DBInstances[0].DBInstanceStatus",
        "--output", "text",
        "--profile", args.profile,
    ])


parser = argparse.ArgumentParser(description="Manage QuantLab RDS instance")
parser.add_argument("--instance-id", default="quantlab-db")
parser.add_argument("--profile", default="quant-lab")

sub = parser.add_subparsers(dest="command", required=True)
sub.add_parser("start", help="Start the instance").set_defaults(func=start)
sub.add_parser("stop", help="Stop the instance").set_defaults(func=stop)
sub.add_parser("status", help="Check instance status").set_defaults(func=status)

if __name__ == "__main__":
    args = parser.parse_args()
    args.func(args)

python rds.py start
python rds.py stop --instance-id my-other-db
python rds.py status --profile prod

Other Patterns

Mutually exclusive

import argparse

parser = argparse.ArgumentParser()
group = parser.add_mutually_exclusive_group()
group.add_argument("--json", action="store_true", help="Output as JSON")
group.add_argument("--csv", action="store_true", help="Output as CSV")
args = parser.parse_args()

# python script.py --json --csv
# error: argument --csv: not allowed with argument --json

fmt = "json" if args.json else "csv" if args.csv else "table"
print(f"Output format: {fmt}")

Default from env var

import argparse
import os

parser = argparse.ArgumentParser()
parser.add_argument(
    "--api-key",
    default=os.environ.get("FRED_API_KEY"),
    help="FRED API key (default: $FRED_API_KEY)",
)
parser.add_argument(
    "--db-url",
    default=os.environ.get("DB_URL", "sqlite:///local.db"),
    help="Database URL (default: $DB_URL or sqlite:///local.db)",
)
args = parser.parse_args()
print(f"api_key={args.api_key}, db_url={args.db_url}")
