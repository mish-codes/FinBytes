Why argparse Over sys.argv

You can always parse command-line arguments with sys.argv. It is a list of strings. But the moment you need type conversion, validation, help text, or optional flags, you are reimplementing argparse badly. Use argparse from the start—it is in the standard library and costs zero dependencies.

# The sys.argv way (don't do this for anything non-trivial)
import sys
ticker = sys.argv[1]           # no validation, no help text
days = int(sys.argv[2])        # crashes with IndexError if missing

# The argparse way
import argparse
parser = argparse.ArgumentParser(description="Fetch historical prices")
parser.add_argument("ticker", help="Stock ticker symbol (e.g., AAPL)")
parser.add_argument("days", type=int, help="Number of trading days")
args = parser.parse_args()
# args.ticker, args.days — validated, typed, with --help for free

Basic Pattern

import argparse

parser = argparse.ArgumentParser(
    description="Bond pricing CLI tool",
    formatter_class=argparse.RawDescriptionHelpFormatter,
)

# Positional argument (required, no --)
parser.add_argument("bond_id", help="CUSIP or ISIN of the bond")

# Optional arguments (--flag)
parser.add_argument("--curve-date", default="today",
                    help="Yield curve date (default: today)")
parser.add_argument("--spread", type=float, default=0.0,
                    help="Spread in basis points (default: 0)")
parser.add_argument("--output", "-o", choices=["json", "csv", "table"],
                    default="table", help="Output format")

args = parser.parse_args()
print(f"Pricing {args.bond_id} as of {args.curve_date}")
print(f"Spread: {args.spread} bps, format: {args.output}")

# Usage
python price_bond.py US912828ZT09 --spread 50 --output json
python price_bond.py --help

Types and Validation

The type parameter converts the string argument to the specified type. If conversion fails, argparse prints an error and exits—no try/except needed in your code.

from pathlib import Path

parser.add_argument("--notional", type=float, help="Notional amount")
parser.add_argument("--count", type=int, help="Number of simulations")
parser.add_argument("--config", type=Path, help="Path to config file")

choices restricts the value to an explicit set:

parser.add_argument("--side", choices=["buy", "sell"],
                    help="Trade side")
parser.add_argument("--env", choices=["dev", "staging", "prod"],
                    default="dev", help="Target environment")

Boolean Flags: The type=bool Trap

This is the single most common argparse mistake. Do not use type=bool:

# WRONG — "False" is a non-empty string, which is truthy!
parser.add_argument("--verbose", type=bool, default=False)
# python script.py --verbose False  →  args.verbose = True  (!)

# CORRECT — use store_true / store_false
parser.add_argument("--verbose", "-v", action="store_true",
                    help="Enable verbose output")
parser.add_argument("--no-cache", action="store_true",
                    help="Bypass cache and fetch fresh data")
# python script.py --verbose        →  args.verbose = True
# python script.py                  →  args.verbose = False

Subcommands

For CLI tools with multiple modes, use add_subparsers. This is how tools like git, docker, and the QuantLab rds.py script work.

"""rds.py — Manage the QuantLab RDS instance (Exercise 11 pattern)"""
import argparse
import subprocess


def start(args):
    print(f"Starting RDS instance: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "start-db-instance",
        "--db-instance-identifier", args.instance_id,
        "--profile", args.profile,
    ])


def stop(args):
    print(f"Stopping RDS instance: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "stop-db-instance",
        "--db-instance-identifier", args.instance_id,
        "--profile", args.profile,
    ])


def status(args):
    print(f"Checking status of: {args.instance_id}")
    subprocess.run([
        "aws", "rds", "describe-db-instances",
        "--db-instance-identifier", args.instance_id,
        "--query", "DBInstances[0].DBInstanceStatus",
        "--output", "text",
        "--profile", args.profile,
    ])


parser = argparse.ArgumentParser(description="Manage QuantLab RDS instance")
parser.add_argument("--instance-id", default="quantlab-db",
                    help="RDS instance identifier")
parser.add_argument("--profile", default="quant-lab",
                    help="AWS CLI profile name")

subparsers = parser.add_subparsers(dest="command", required=True)

sub_start = subparsers.add_parser("start", help="Start the instance")
sub_start.set_defaults(func=start)

sub_stop = subparsers.add_parser("stop", help="Stop the instance")
sub_stop.set_defaults(func=stop)

sub_status = subparsers.add_parser("status", help="Check instance status")
sub_status.set_defaults(func=status)

args = parser.parse_args()
args.func(args)

# Usage
python rds.py start
python rds.py stop --instance-id my-other-db
python rds.py status --profile prod
python rds.py --help
python rds.py start --help

Mutual Exclusion

When two flags cannot coexist (e.g., --json and --csv):

group = parser.add_mutually_exclusive_group()
group.add_argument("--json", action="store_true", help="Output as JSON")
group.add_argument("--csv", action="store_true", help="Output as CSV")

# python script.py --json --csv
# error: argument --csv: not allowed with argument --json

The if __name__ Pattern

Always guard the parser behind if __name__ == "__main__" so the module can be imported without triggering argument parsing:

def main():
    parser = argparse.ArgumentParser(description="Yield curve tool")
    parser.add_argument("curve_date", help="YYYY-MM-DD")
    parser.add_argument("--plot", action="store_true")
    args = parser.parse_args()

    curve = fetch_curve(args.curve_date)
    if args.plot:
        plot_curve(curve)
    else:
        print(curve)


if __name__ == "__main__":
    main()

Alternatives: click and typer

click uses decorators instead of an imperative API. It handles colors, prompts, and file arguments more cleanly than argparse. It is a dependency (pip install click) but is already pulled in by Flask.

typer builds on click and uses type hints to define arguments. The least boilerplate of all three. It is a good choice for new projects where you control dependencies.

# typer equivalent of the basic argparse example
import typer

app = typer.Typer()

@app.command()
def price(
    bond_id: str,
    curve_date: str = "today",
    spread: float = 0.0,
    output: str = typer.Option("table", help="json, csv, or table"),
):
    """Price a bond given a curve date and spread."""
    print(f"Pricing {bond_id} as of {curve_date}, spread={spread}")

if __name__ == "__main__":
    app()

When to prefer argparse: Standard library only, no external deps, simple scripts. When to prefer click/typer: Larger CLI tools, multiple commands, richer UX (colors, progress bars, prompts).
