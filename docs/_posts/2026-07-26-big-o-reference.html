Common mistakes

Searching a list when you should use a set

# O(n) per check — 1M items × 1K checks = 1 billion operations
watchlist = ["AAPL", "MSFT", "GOOG", ...]  # list
if "TSLA" in watchlist:  # scans entire list
    ...

# O(1) per check — use a set
watchlist = {"AAPL", "MSFT", "GOOG", ...}  # set
if "TSLA" in watchlist:  # hash lookup
    ...

Appending to the front of a list

from collections import deque

# O(n) — shifts every element right
prices = []
prices.insert(0, 189.50)  # slow at scale

# O(1) — use deque
prices = deque(maxlen=1000)
prices.appendleft(189.50)  # always fast

Nested loop when you can use a dict

# O(n²) — matching trades to positions
for trade in trades:          # n trades
    for pos in positions:     # m positions
        if trade.ticker == pos.ticker:
            pos.update(trade)

# O(n) — index positions by ticker first
pos_map = {p.ticker: p for p in positions}  # O(m)
for trade in trades:                         # O(n)
    if trade.ticker in pos_map:              # O(1)
        pos_map[trade.ticker].update(trade)

Sorting when you only need min/max

import heapq

# O(n log n) — sorts everything to find top 5
top5 = sorted(returns, reverse=True)[:5]

# O(n log k) — heap only tracks top k
top5 = heapq.nlargest(5, returns)

# O(n) — just need the one biggest/smallest
worst = min(returns)
best = max(returns)
