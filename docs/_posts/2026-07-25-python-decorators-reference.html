What a decorator does

A decorator wraps a function. @decorator above a function is syntactic sugar for func = decorator(func).

def log_call(func):
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} returned {result}")
        return result
    return wrapper

@log_call
def get_price(ticker: str) -> float:
    return 189.50

get_price("AAPL")
# Calling get_price
# get_price returned 189.5

Preserve the original function's identity

from functools import wraps

def log_call(func):
    @wraps(func)  # preserves __name__, __doc__, __module__
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

Without @wraps, get_price.__name__ returns "wrapper" and help/docs break.

Decorator with arguments

from functools import wraps

def retry(max_attempts: int = 3):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts:
                        raise
                    print(f"Attempt {attempt} failed: {e}")
        return wrapper
    return decorator

@retry(max_attempts=5)
def fetch_price(ticker: str) -> float:
    import random
    if random.random() < 0.7:
        raise ConnectionError("API timeout")
    return 189.50

Timer decorator

import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

@timer
def bootstrap_curve(par_yields: list[float]) -> list[float]:
    time.sleep(0.1)  # simulate work
    return [y + 0.02 for y in par_yields]

Cache decorator

from functools import lru_cache

@lru_cache(maxsize=128)
def get_spot_rate(tenor: str) -> float:
    """Expensive DB lookup — cached after first call."""
    print(f"Fetching {tenor} from database...")
    return {"1Y": 4.50, "5Y": 4.90, "10Y": 5.00}[tenor]

get_spot_rate("5Y")  # Fetching 5Y from database...
get_spot_rate("5Y")  # instant — cached

Validation decorator

from functools import wraps

def validate_ticker(func):
    @wraps(func)
    def wrapper(ticker: str, *args, **kwargs):
        if not ticker.isalpha() or len(ticker) > 5:
            raise ValueError(f"Invalid ticker: {ticker}")
        return func(ticker.upper(), *args, **kwargs)
    return wrapper

@validate_ticker
def get_price(ticker: str) -> float:
    return 189.50

get_price("aapl")  # works — converted to AAPL
get_price("12345!")  # ValueError: Invalid ticker: 12345!

Class-based decorator

class CountCalls:
    def __init__(self, func):
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        return self.func(*args, **kwargs)

@CountCalls
def price_bond(face: float, rate: float, years: int) -> float:
    return face / (1 + rate) ** years

price_bond(100, 0.05, 5)
price_bond(100, 0.06, 10)
print(price_bond.count)  # 2

Stacking decorators

@timer
@retry(max_attempts=3)
@validate_ticker
def fetch_and_price(ticker: str) -> float:
    return get_price(ticker)

# Execution order (bottom to top):
# validate_ticker runs first → retry wraps that → timer wraps that
