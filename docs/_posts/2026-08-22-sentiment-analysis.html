Why sentiment matters in finance

Markets move on information and emotion. Sentiment analysis quantifies the emotional tone of text, turning headlines, earnings calls, and social media posts into numbers you can backtest against. A negative sentiment shift in news coverage often precedes price drops by hours or days.

TextBlob sentiment

from textblob import TextBlob
import pandas as pd

def textblob_sentiment(text: str) -> dict:
    """Return polarity (-1 to 1) and subjectivity (0 to 1)."""
    blob = TextBlob(text)
    return {
        "polarity": round(blob.sentiment.polarity, 4),
        "subjectivity": round(blob.sentiment.subjectivity, 4),
    }

tb_results = pd.DataFrame([
    {"headline": h, **textblob_sentiment(h)} for h in headlines
])
print(tb_results[["headline", "polarity"]].to_string(index=False))

TextBlob uses a pattern-based approach. It looks up words in a pre-built lexicon and averages their scores. It is simple and fast but struggles with financial jargon. "Rates tumble" is negative in general text, but "yields tumble" might be positive for bond holders.

VADER sentiment

from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

analyzer = SentimentIntensityAnalyzer()

def vader_sentiment(text: str) -> dict:
    """Return VADER compound score and component scores."""
    scores = analyzer.polarity_scores(text)
    return {
        "compound": scores["compound"],
        "positive": scores["pos"],
        "negative": scores["neg"],
        "neutral": scores["neu"],
    }

vader_results = pd.DataFrame([
    {"headline": h, **vader_sentiment(h)} for h in headlines
])
print(vader_results[["headline", "compound"]].to_string(index=False))

VADER (Valence Aware Dictionary and sEntiment Reasoner) is better suited for short texts. It handles capitalization, punctuation, and intensifiers ("very", "extremely"). The compound score ranges from -1 (most negative) to +1 (most positive). A common threshold: compound >= 0.05 is positive, <= -0.05 is negative, otherwise neutral.

Compare the two

comparison = pd.DataFrame({
    "headline": headlines,
    "textblob": tb_results["polarity"],
    "vader": vader_results["compound"],
})

comparison["agree"] = (
    (comparison["textblob"] > 0) & (comparison["vader"] > 0.05) |
    (comparison["textblob"] < 0) & (comparison["vader"] < -0.05) |
    (comparison["textblob"] == 0) & (comparison["vader"].between(-0.05, 0.05))
)

agreement_pct = comparison["agree"].mean() * 100
print(f"\nAgreement rate: {agreement_pct:.0f}%")
print(comparison[["headline", "textblob", "vader", "agree"]].to_string(index=False))

Sentiment distribution chart

import plotly.graph_objects as go

fig = make_subplots(rows=1, cols=2, subplot_titles=["TextBlob Polarity", "VADER Compound"])

fig.add_trace(go.Histogram(
    x=tb_results["polarity"], nbinsx=12,
    marker_color="steelblue", opacity=0.7, name="TextBlob"

                  title="Sentiment Score Distributions", showlegend=False)

Sentiment over time

import numpy as np

# Assign dates to headlines (simulate a daily feed)
dates = pd.date_range("2026-01-01", periods=len(headlines), freq="B")

timeline = pd.DataFrame({
    "date": dates,
    "vader": vader_results["compound"],
    "textblob": tb_results["polarity"],
})
timeline["vader_rolling"] = timeline["vader"].rolling(3, min_periods=1).mean()

fig = go.Figure()
fig.add_trace(go.Bar(x=timeline["date"], y=timeline["vader"],
                     marker_color=["green" if v > 0.05 else "red" if v < -0.05 else "gray"
                                   for v in timeline["vader"]],
                     name="VADER daily"))
fig.add_trace(go.Scatter(x=timeline["date"], y=timeline["vader_rolling"],
                         mode="lines", line=dict(color="black", width=2),
                         name="3-day rolling avg"))

fig.update_layout(height=400, width=800, template="plotly_white",
                  title="Sentiment Timeline", xaxis_title="Date",
                  yaxis_title="VADER Compound Score")
fig.show()

Word frequency (toward a word cloud)

from collections import Counter

# Simple word frequency from headlines
words = " ".join(headlines).lower().split()
stopwords = {"the", "a", "an", "of", "on", "in", "to", "for", "as", "with", "from", "by", "after"}
filtered = [w.strip(",.!?") for w in words if w.strip(",.!?") not in stopwords and len(w) > 2]

word_counts = Counter(filtered).most_common(15)
words_df = pd.DataFrame(word_counts, columns=["word", "count"])

fig = go.Figure(go.Bar(
    x=words_df["count"], y=words_df["word"],
    orientation="h", marker_color="steelblue"
))
fig.update_layout(height=400, width=600, template="plotly_white",
                  title="Top Words in Headlines", yaxis=dict(autorange="reversed"))
fig.show()

For a proper word cloud, use the wordcloud library: WordCloud().generate(text). The bar chart above serves the same analytical purpose and is easier to read precisely.

Limitations

  • Domain gap — general-purpose sentiment tools misread financial language. "Volatile" is neutral in finance but negative in general text. Consider fine-tuning on financial corpora (e.g., FinBERT).
  • Sarcasm and context — "Great, another rate hike" reads as positive to a lexicon-based tool.
  • Lag — by the time sentiment is measurable in news, the move may already be priced in.
