The approach

A budget tracker boils down to three things: what comes in, what goes out, and what is left. The code below structures income and expenses as simple dictionaries, computes the surplus or deficit, and generates a spending breakdown chart.

Budget calculator

from dataclasses import dataclass, field


@dataclass
class Budget:
    """Monthly personal budget."""
    income: dict[str, float] = field(default_factory=dict)
    expenses: dict[str, float] = field(default_factory=dict)

    @property
    def total_income(self) -> float:
        return round(sum(self.income.values()), 2)

    @property
    def total_expenses(self) -> float:
        return round(sum(self.expenses.values()), 2)

    @property
    def surplus(self) -> float:
        return round(self.total_income - self.total_expenses, 2)

    @property
    def savings_rate(self) -> float:
        """Fraction of income saved (0-1). Negative means deficit."""
        if self.total_income == 0:
            return 0.0
        return round(self.surplus / self.total_income, 4)

    def summary(self) -> str:
        status = "SURPLUS" if self.surplus >= 0 else "DEFICIT"
        lines = [
            f"Total income   : ${self.total_income:,.2f}",
            f"Total expenses : ${self.total_expenses:,.2f}",
            f"{status:14s} : ${abs(self.surplus):,.2f}",
            f"Savings rate   : {self.savings_rate:.1%}",
        ]
        return "\n".join(lines)

budget = Budget(

Spending pie chart

import plotly.graph_objects as go

budget = Budget(
    income={"Salary": 5_500, "Freelance": 800},
    expenses={
        "Rent": 1_600, "Groceries": 450, "Utilities": 180,
        "Transport": 250, "Insurance": 320, "Subscriptions": 65,
        "Dining out": 200, "Entertainment": 120,
        "Savings / invest": 1_000, "Miscellaneous": 150,
    },
)

categories = list(budget.expenses.keys()) + ["Unallocated surplus"]
values = list(budget.expenses.values()) + [budget.surplus]

fig = go.Figure(go.Pie(
    labels=categories,
    values=values,
    hole=0.4,
    textinfo="label+percent",
    textposition="outside",
    marker=dict(colors=[
        "#636EFA", "#EF553B", "#00CC96", "#AB63FA",
        "#FFA15A", "#19D3F3", "#FF6692", "#B6E880",
        "#FF97FF", "#FECB52", "#72B7B2",
    ]),
))
fig.update_layout(
    title="Monthly Spending Breakdown",
    template="plotly_dark",
    margin=dict(t=50, b=20, l=20, r=20),
    showlegend=False,
)
fig.show()

Key takeaways

  • The 50/30/20 rule suggests 50 % needs, 30 % wants, 20 % savings. This budget allocates ~47 % to needs, ~8 % to wants, and ~47 % to savings + surplus — well ahead of target.
  • Track savings_rate monthly. A consistent 15 %+ rate is the single best predictor of long-term financial health.
  • Swap the hardcoded dicts for a CSV or database to build a real tracker that persists across months.
