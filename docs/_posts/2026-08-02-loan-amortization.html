The PMT formula

A fixed-rate loan has a constant monthly payment calculated from:

PMT = P * r * (1 + r)^n / ((1 + r)^n - 1)

P = principal (loan amount)
r = monthly interest rate (annual_rate / 12)
n = total number of monthly payments

Each payment is split between interest (charged on the remaining balance) and principal repayment. Early payments are interest-heavy; later ones are principal-heavy.

Schedule generator

from dataclasses import dataclass


@dataclass
class AmortRow:
    month: int
    payment: float
    principal: float
    interest: float
    balance: float


def pmt(principal: float, annual_rate: float, years: int) -> float:
    """Monthly payment for a fixed-rate loan."""
    r = annual_rate / 12
    n = years * 12
    if r == 0:
        return round(principal / n, 2)
    return round(principal * r * (1 + r) ** n / ((1 + r) ** n - 1), 2)


def amortization_schedule(
    principal: float,
    annual_rate: float,
    years: int,
) -> list[AmortRow]:
    """Full month-by-month amortization schedule."""
    r = annual_rate / 12
    n = years * 12
    monthly = pmt(principal, annual_rate, years)
    balance = principal
    schedule: list[AmortRow] = []

    for month in range(1, n + 1):
        interest = round(balance * r, 2)
        princ = round(monthly - interest, 2)
        if month == n:            # final payment absorbs rounding
            princ = round(balance, 2)
            monthly = princ + interest
        balance = round(balance - princ, 2)
        schedule.append(AmortRow(month, monthly, princ, interest, balance))

    return schedule


# --- demo ---
rows = amortization_schedule(
    principal=300_000,
    annual_rate=0.065,
    years=30,
)

total_interest = sum(r.interest for r in rows)
print(f"Monthly payment : ${rows[0].payment:,.2f}")
print(f"Total interest  : ${total_interest:,.2f}")
print(f"Total cost      : ${sum(r.payment for r in rows):,.2f}")

Principal vs interest stacked area chart

import plotly.graph_objects as go

rows = amortization_schedule(300_000, 0.065, 30)

months = [r.month for r in rows]
principals = [r.principal for r in rows]
interests = [r.interest for r in rows]

fig = go.Figure()
fig.add_trace(go.Scatter(
    x=months, y=principals,
    mode="lines", name="Principal",
    stackgroup="one",
    line=dict(color="#636EFA"),
))
fig.add_trace(go.Scatter(
    x=months, y=interests,
    mode="lines", name="Interest",
    stackgroup="one",
    line=dict(color="#EF553B"),
))
fig.update_layout(
    title="Monthly Payment Split: Principal vs Interest",
    xaxis_title="Month",
    yaxis_title="Dollars ($)",
    template="plotly_dark",
    margin=dict(t=50, b=40, l=60, r=20),
)
fig.show()

Key takeaways

  • On a 30-year mortgage at 6.5 %, you pay roughly $383,000 in interest on top of the $300,000 principal.
  • The crossover point (where principal exceeds interest) does not arrive until roughly month 218 — over 18 years in.
  • Even small extra payments early on shave months off the schedule because they attack the principal directly.
